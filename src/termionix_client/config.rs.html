17//! Client configuration
18
19use std::time::Duration;
20
21/// Telnet client configuration
22#[derive(Debug, Clone)]
23pub struct ClientConfig {
24    /// Server hostname or IP address
25    pub host: String,
26
27    /// Server port
28    pub port: u16,
29
30    /// Terminal type to report (e.g., "xterm-256color")
31    pub terminal_type: String,
32
33    /// Terminal width in columns
34    pub terminal_width: u16,
35
36    /// Terminal height in rows
37    pub terminal_height: u16,
38
39    /// Connection timeout
40    pub connect_timeout: Duration,
41
42    /// Read timeout (None for no timeout)
43    pub read_timeout: Option<Duration>,
44
45    /// Enable automatic reconnection on logout.txt
46    pub auto_reconnect: bool,
47
48    /// Delay before reconnection attempt
49    pub reconnect_delay: Duration,
50
51    /// Maximum number of reconnection attempts (None for unlimited)
52    pub max_reconnect_attempts: Option<usize>,
53
54    /// Buffer size for incoming data
55    pub buffer_size: usize,
56
57    /// Enable keepalive
58    pub keepalive: bool,
59
60    /// Keepalive interval
61    pub keepalive_interval: Duration,
62}
63
64impl Default for ClientConfig {
65    fn default() -> Self {
66        Self {
67            host: "localhost".to_string(),
68            port: 23,
69            terminal_type: "xterm-256color".to_string(),
70            terminal_width: 80,
71            terminal_height: 24,
72            connect_timeout: Duration::from_secs(10),
73            read_timeout: Some(Duration::from_secs(300)), // 5 minutes
74            auto_reconnect: false,
75            reconnect_delay: Duration::from_secs(5),
76            max_reconnect_attempts: Some(3),
77            buffer_size: 8192,
78            keepalive: true,
79            keepalive_interval: Duration::from_secs(60),
80        }
81    }
82}
83
84impl ClientConfig {
85    /// Create a new client configuration with the given host and port
86    pub fn new(host: impl Into<String>, port: u16) -> Self {
87        Self {
88            host: host.into(),
89            port,
90            ..Default::default()
91        }
92    }
93
94    /// Set the terminal type
95    pub fn with_terminal_type(mut self, terminal_type: impl Into<String>) -> Self {
96        self.terminal_type = terminal_type.into();
97        self
98    }
99
100    /// Set the terminal size
101    pub fn with_terminal_size(mut self, width: u16, height: u16) -> Self {
102        self.terminal_width = width;
103        self.terminal_height = height;
104        self
105    }
106
107    /// Set the connection timeout
108    pub fn with_connect_timeout(mut self, timeout: Duration) -> Self {
109        self.connect_timeout = timeout;
110        self
111    }
112
113    /// Set the read timeout
114    pub fn with_read_timeout(mut self, timeout: Option<Duration>) -> Self {
115        self.read_timeout = timeout;
116        self
117    }
118
119    /// Enable automatic reconnection
120    pub fn with_auto_reconnect(mut self, enabled: bool) -> Self {
121        self.auto_reconnect = enabled;
122        self
123    }
124
125    /// Set the reconnection delay
126    pub fn with_reconnect_delay(mut self, delay: Duration) -> Self {
127        self.reconnect_delay = delay;
128        self
129    }
130
131    /// Set the maximum reconnection attempts
132    pub fn with_max_reconnect_attempts(mut self, max: Option<usize>) -> Self {
133        self.max_reconnect_attempts = max;
134        self
135    }
136
137    /// Get the server address as a string
138    pub fn address(&self) -> String {
139        format!("{}:{}", self.host, self.port)
140    }
141}
