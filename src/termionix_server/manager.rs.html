17//! Connection manager implementation
18//!
19//! The ConnectionManager is responsible for:
20//! - Managing all active connections
21//! - Spawning and tracking connection workers
22//! - Broadcasting messages to all connections
23//! - Graceful shutdown coordination
24//! - Connection lifecycle tracking

39/// Result of a broadcast operation
40#[derive(Debug, Clone)]
41pub struct BroadcastResult {
42    /// Total number of connections attempted
43    pub total: usize,
44    /// Number of successful sends
45    pub succeeded: usize,
46    /// Number of failed sends
47    pub failed: usize,
48    /// Errors that occurred (ConnectionId and error message)
49    pub errors: Vec<(ConnectionId, String)>,
50}
51
52impl BroadcastResult {
53    /// Create a new empty result
54    fn new() -> Self {
55        Self {
56            total: 0,
57            succeeded: 0,
58            failed: 0,
59            errors: Vec::new(),
60        }
61    }
62
63    /// Check if all broadcasts succeeded
64    pub fn all_succeeded(&self) -> bool {
65        self.failed == 0
66    }
67
68    /// Get the success rate as a percentage
69    pub fn success_rate(&self) -> f64 {
70        if self.total == 0 {
71            100.0
72        } else {
73            (self.succeeded as f64 / self.total as f64) * 100.0
74        }
75    }
76}

78/// Managed connection entry
79struct ManagedConnection {
80    /// Connection ID
81    id: ConnectionId,
82    /// The connection itself
83    connection: TelnetConnection,
84    /// Control channel sender
85    control_tx: mpsc::Sender<ControlMessage>,
86    /// Worker task handle
87    worker_handle: JoinHandle<()>,
88    /// Current state (atomic for lock-free access)
89    state: Arc<std::sync::atomic::AtomicU8>,
90    /// When the connection was created
91    created_at: Instant,
92}
93
94impl ManagedConnection {
95    /// Get the current state
96    fn state(&self) -> ConnectionState {
97        ConnectionState::from_u8(self.state.load(Ordering::Acquire))
98    }
99
100    /// Get connection info snapshot
101    fn info(&self) -> ConnectionInfo {
102        ConnectionInfo {
103            id: self.id,
104            state: self.state(),
105            peer_addr: self.connection.peer_addr(),
106            created_at: self.created_at,
107            last_activity: self.created_at, // Worker tracks this internally
108            bytes_sent: self.connection.bytes_sent(),
109            bytes_received: self.connection.bytes_received(),
110            messages_sent: self.connection.messages_sent(),
111            messages_received: self.connection.messages_received(),
112        }
113    }
114}

116/// Connection manager
117pub struct ConnectionManager {
118    /// Active connections (lock-free concurrent map)
119    connections: Arc<DashMap<ConnectionId, ManagedConnection>>,
120    /// Next connection ID (monotonically increasing)
121    next_id: Arc<AtomicU64>,
122    /// Server metrics
123    metrics: Arc<ServerMetrics>,
124    /// Worker configuration
125    worker_config: WorkerConfig,
126}
127
128impl ConnectionManager {
129    /// Create a new connection manager
130    pub fn new(metrics: Arc<ServerMetrics>, worker_config: WorkerConfig) -> Self {
131        Self {
132            connections: Arc::new(DashMap::new()),
133            next_id: Arc::new(AtomicU64::new(1)),
134            metrics,
135            worker_config,
136        }
137    }
138
139    /// Get the next connection ID
140    fn next_connection_id(&self) -> ConnectionId {
141        let id = self.next_id.fetch_add(1, Ordering::SeqCst);
142        ConnectionId::new(id)
143    }
144
145    /// Add a new connection
146    ///
147    /// This spawns a worker task for the connection and tracks it.
148    #[instrument(skip(self, connection, handler), fields(id))]
149    pub fn add_connection(
150        &self,
151        connection: TelnetConnection,
152        handler: Arc<dyn ServerHandler>,
153    ) -> Result<ConnectionId> {
154        let id = self.next_connection_id();
155        tracing::Span::current().record("id", id.to_string());
156
157        // Create shared state
158        let state = Arc::new(std::sync::atomic::AtomicU8::new(
159            ConnectionState::Connecting.as_u8(),
160        ));
161        let worker_state = state.clone();
162
163        // Create worker
164        let worker_connection = connection.clone();
165        let (worker, control_tx) = crate::ConnectionWorker::new(
166            id,
167            worker_connection,
168            handler,
169            self.worker_config.clone(),
170            worker_state,
171        );
172
173        // Spawn worker task
174        let connections = self.connections.clone();
175        let metrics = self.metrics.clone();
176        let worker_handle = tokio::spawn(async move {
177            let start = Instant::now();
178            worker.run().await;
179
180            // Cleanup after worker finishes
181            connections.remove(&id);
182            metrics.connection_closed(start.elapsed());
183        });
184
185        // Store managed connection
186        let managed = ManagedConnection {
187            id,
188            connection,
189            control_tx,
190            worker_handle,
191            state,
192            created_at: Instant::now(),
193        };
194
195        self.connections.insert(id, managed);
196        self.metrics.connection_opened();
197
198        Ok(id)
199    }

201    /// Remove a connection
202    ///
203    /// This sends a close message to the worker and removes it from tracking.
204    pub async fn remove_connection(&self, id: ConnectionId) -> Result<()> {
205        if let Some((_, managed)) = self.connections.remove(&id) {
206            // Send close message (best effort)
207            let _ = managed.control_tx.send(ControlMessage::Close).await;
208
209            // Wait for worker to finish (with timeout)
210            let _ = tokio::time::timeout(std::time::Duration::from_secs(5), managed.worker_handle)
211                .await;
212
213            Ok(())
214        } else {
215            Err(TelnetError::ConnectionNotFound(id))
216        }
217    }

244    /// Get the number of active connections
245    pub fn connection_count(&self) -> usize {
246        self.connections.len()
247    }

249    /// Send a command to a specific connection
250    pub async fn send_to_connection(
251        &self,
252        id: ConnectionId,
253        command: TerminalCommand,
254    ) -> Result<()> {
255        if let Some(managed) = self.connections.get(&id) {
256            managed
257                .control_tx
258                .send(ControlMessage::SendCommand(command))
259                .await
260                .map_err(|_| TelnetError::ConnectionClosed)?;
261            Ok(())
262        } else {
263            Err(TelnetError::ConnectionNotFound(id))
264        }
265    }

267    /// Broadcast a command to all connections
268    ///
269    /// This sends the command to all active connections concurrently.
270    /// Returns a result with statistics about the broadcast.
271    #[instrument(skip(self))]
272    pub async fn broadcast(&self, command: TerminalCommand) -> BroadcastResult {
273        use futures_util::stream::{self, StreamExt};
274
275        let mut result = BroadcastResult::new();
276        result.total = self.connections.len();
277
278        // Create stream of send operations
279        let sends = self.connections.iter().map(|entry| {
280            let id = *entry.key();
281            let tx = entry.control_tx.clone();
282            let cmd = command.clone();
283
284            async move {
285                match tx.send(ControlMessage::Broadcast(cmd)).await {
286                    Ok(_) => (id, Ok(())),
287                    Err(e) => (id, Err(e.to_string())),
288                }
289            }
290        });
291
292        // Process in batches of 100 concurrent operations to avoid memory spike
293        let mut results_stream = stream::iter(sends).buffer_unordered(100);
294
295        while let Some((id, res)) = results_stream.next().await {
296            match res {
297                Ok(_) => result.succeeded += 1,
298                Err(e) => {
299                    result.failed += 1;
300                    result.errors.push((id, e));
301                }
302            }
303        }
304
305        result
306    }
307
308    /// Broadcast to connections matching a filter
309    pub async fn broadcast_filtered<F>(
310        &self,
311        command: TerminalCommand,
312        filter: F,
313    ) -> BroadcastResult
314    where
315        F: Fn(&ConnectionInfo) -> bool,
316    {
317        let mut result = BroadcastResult::new();
318
319        // Collect matching connections
320        let mut sends = Vec::new();
321        for entry in self.connections.iter() {
322            let info = entry.value().info();
323            if filter(&info) {
324                result.total += 1;
325                let id = *entry.key();
326                let tx = entry.control_tx.clone();
327                let cmd = command.clone();
328
329                sends.push(async move {
330                    match tx.send(ControlMessage::Broadcast(cmd)).await {
331                        Ok(_) => (id, Ok(())),
332                        Err(e) => (id, Err(e.to_string())),
333                    }
334                });
335            }
336        }
337
338        // Execute all sends concurrently
339        let results = futures_util::future::join_all(sends).await;
340
341        // Collect results
342        for (id, res) in results {
343            match res {
344                Ok(_) => result.succeeded += 1,
345                Err(e) => {
346                    result.failed += 1;
347                    result.errors.push((id, e));
348                }
349            }
350        }
351
352        result
353    }

355    /// Broadcast to all connections except specified ones
356    ///
357    /// This is useful for broadcasting messages to all connections except
358    /// the sender or a specific set of connections.
359    ///
360    /// # Parameters
361    ///
362    /// * `command` - The terminal command to broadcast
363    /// * `exclude` - Slice of connection IDs to exclude from the broadcast
364    ///
365    /// # Returns
366    ///
367    /// A `BroadcastResult` containing statistics about the broadcast operation.
368    ///
369    /// # Example
370    ///
371    /// ```no_run
372    /// # use termionix_server::ConnectionManager;
373    /// # use termionix_terminal::TerminalCommand;
374    /// # async fn example(manager: &ConnectionManager, sender_id: termionix_server::ConnectionId) {
375    /// // Broadcast to all except the sender
376    /// let result = manager.broadcast_except(
377    ///     TerminalCommand::SendEraseLine,
378    ///     &[sender_id]
379    /// ).await;
380    /// println!("Broadcast to {} connections", result.succeeded);
381    /// # }
382    /// ```
383    pub async fn broadcast_except(
384        &self,
385        command: TerminalCommand,
386        exclude: &[ConnectionId],
387    ) -> BroadcastResult {
388        let mut result = BroadcastResult::new();
389
390        // Collect non-excluded connections
391        let mut sends = Vec::new();
392        for entry in self.connections.iter() {
393            let id = *entry.key();
394
395            // Skip excluded connections
396            if exclude.contains(&id) {
397                continue;
398            }
399
400            result.total += 1;
401            let tx = entry.control_tx.clone();
402            let cmd = command.clone();
403
404            sends.push(async move {
405                match tx.send(ControlMessage::Broadcast(cmd)).await {
406                    Ok(_) => (id, Ok(())),
407                    Err(e) => (id, Err(e.to_string())),
408                }
409            });
410        }
411
412        // Execute all sends concurrently
413        let results = futures_util::future::join_all(sends).await;
414
415        // Collect results
416        for (id, res) in results {
417            match res {
418                Ok(_) => result.succeeded += 1,
419                Err(e) => {
420                    result.failed += 1;
421                    result.errors.push((id, e));
422                }
423            }
424        }
425
426        result
427    }
428
429    /// Shutdown all connections gracefully
430    pub async fn shutdown(&self) {
431        // Send close to all connections
432        for entry in self.connections.iter() {
433            let _ = entry.control_tx.send(ControlMessage::Close).await;
434        }
435
436        // Give workers a brief moment to process close messages
437        // This is much faster than the previous 2-second sleep
438        tokio::time::sleep(std::time::Duration::from_millis(50)).await;
439
440        // Abort any remaining workers
441        for entry in self.connections.iter() {
442            entry.worker_handle.abort();
443        }
444
445        // Clear all connections
446        self.connections.clear();
447    }
448}

450impl std::fmt::Debug for ConnectionManager {
451    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

516        let worker_config = WorkerConfig {
517            read_timeout: config.read_timeout,
518            idle_timeout: config.idle_timeout,

522        let manager = ConnectionManager::new(metrics.clone(), worker_config);
523
524        // Add multiple connections
525        let mut clients = Vec::new();
526        for i in 0..3 {
527            let (server, client) = create_test_connection().await;
528            let id = ConnectionId::new(i);
529            let connection = TelnetConnection::wrap(server, id).unwrap();
530            manager
531                .add_connection(connection, Arc::new(TestHandler))

533            clients.push(client);

538        // Broadcast a command
539        let result = manager.broadcast(TerminalCommand::EraseLine).await;
540        assert_eq!(result.total, 3);
541
542        // Cleanup
543        manager.shutdown().await;
544        drop(clients);
545    }
