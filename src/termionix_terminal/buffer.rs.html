23/// Virtual Terminal Buffer
24#[derive(Clone)]
25pub struct TerminalBuffer {
26    /// Terminal dimensions (width, height)
27    size: TerminalSize,

29    cursor: CursorPosition,
30    /// Terminal Environment variables
31    environment: BTreeMap<String, String>,
32    /// Completed lines
33    completed_lines: Vec<SegmentedString>,

57    /// Sets the terminal size
58    pub fn set_size(&mut self, width: usize, height: usize) {
59        self.size = TerminalSize::new(width, height);
60        // Adjust the cursor if it's now out of bounds
61        if self.cursor.col >= self.size.cols {
62            self.cursor.col = self.size.cols.saturating_sub(1);
63        }
64        if self.cursor.row >= self.size.rows {
65            self.cursor.row = self.size.rows.saturating_sub(1);
66        }
67    }

101    /// Returns an iterator over all environment variable key-value pairs.
102    ///
103    /// This method provides read-only access to the terminal's environment variables
104    /// stored in the buffer. The iterator yields references to both keys and values.
105    ///
106    /// # Returns
107    ///
108    /// An iterator that yields tuples of `(&String, &String)` representing
109    /// (key, value) pairs of environment variables.
110    ///
111    /// # Examples
112    ///
113    /// ```rust
114    /// # use termionix_terminal::TerminalBuffer;
115    /// let mut buffer = TerminalBuffer::new();
116    /// buffer.set_environment("USER", "alice");
117    /// buffer.set_environment("SHELL", "/bin/bash");

119    /// for (key, value) in buffer.environment() {
120    ///     println!("{} = {}", key, value);
121    /// }
122    /// ```

124        self.environment.iter()
125    }

128    ///
129    /// This method stores or updates an environment variable key-value pair.
130    /// If the key already exists, its value will be replaced with the new value.
131    /// Both the key and value are converted to owned `String` instances.
132    ///
133    /// # Parameters
134    ///
135    /// * `key` - The environment variable name
136    /// * `value` - The environment variable value

138    /// # Examples

140    /// ```rust
141    /// # use termionix_terminal::TerminalBuffer;
142    /// let mut buffer = TerminalBuffer::new();
143    ///
144    /// // Set a new environment variable
145    /// buffer.set_environment("PATH", "/usr/local/bin:/usr/bin");
146    ///
147    /// // Update an existing environment variable
148    /// buffer.set_environment("PATH", "/opt/bin:/usr/local/bin:/usr/bin");

155    ///
156    /// This method looks up an environment variable in the terminal buffer

159    /// # Parameters

161    /// * `key` - The environment variable name to look up
162    ///
163    /// # Returns
164    ///
165    /// * `Some(&String)` - A reference to the value if the key exists
166    /// * `None` - If the key doesn't exist in the environment

168    /// # Examples

184        self.environment.get(key)

187    /// Moves the cursor by a relative offset in both column and row directions.
188    ///
189    /// This method adjusts the cursor position by the specified column and row deltas,
190    /// with negative values moving left/up and positive values moving right/down.
191    /// The resulting position is automatically clamped to the terminal bounds.
192    ///
193    /// # Parameters
194    ///
195    /// * `col` - The column delta to move the cursor. Positive values move right,
196    ///           negative values move left.
197    /// * `row` - The row delta to move the cursor. Positive values move down,
198    ///           negative values move up.
199    ///
200    /// # Returns
201    ///
202    /// Returns a tuple `(usize, usize)` representing the new cursor position after
203    /// the move operation, where the first element is the column and the second is
204    /// the row. Both values are 0-indexed and clamped to the terminal bounds.
205    ///
206    /// # Bounds Handling
207    ///
208    /// * Column values are clamped to `[0, width - 1]`
209    /// * Row values are clamped to `[0, height - 1]`
210    /// * Operations use saturating arithmetic to prevent overflow/underflow
211    ///
212    /// # Examples
213    ///
214    /// ```rust
215    /// # use termionix_terminal::{CursorPosition, TerminalBuffer};
216    /// let mut buffer = TerminalBuffer::new_with_size(80, 24);
217    /// buffer.set_cursor_position(10, 5);
218    ///
219    /// // Move right 5 columns and down 2 rows
220    /// let pos = buffer.move_cursor(5, 2);
221    /// assert_eq!(pos, CursorPosition::new(15, 7));
222    ///
223    /// // Move left 20 columns (will clamp to 0)
224    /// let pos = buffer.move_cursor(-20, 0);
225    /// assert_eq!(pos, CursorPosition::new(0, 7));
226    ///
227    /// // Move to a position that exceeds bounds (will clamp)
228    /// buffer.set_cursor_position(75, 20);
229    /// let pos = buffer.move_cursor(100, 100);
230    /// assert_eq!(pos, CursorPosition::new(79, 23)); // Clamped to terminal bounds (80x24)
231    /// ```
232    pub fn move_cursor(&mut self, col_delta: isize, row_delta: isize) -> CursorPosition {
233        // Calculate the new column position
234        let new_col = if col_delta >= 0 {
235            self.cursor.col.saturating_add(col_delta as usize)
236        } else {
237            self.cursor.col.saturating_sub(col_delta.unsigned_abs())
238        };
239
240        // Calculate the new row position
241        let new_row = if row_delta >= 0 {
242            self.cursor.row.saturating_add(row_delta as usize)
243        } else {
244            self.cursor.row.saturating_sub(row_delta.unsigned_abs())
245        };
246
247        // Clamp to the terminal bounds
248        self.cursor.col = new_col.min(self.size.cols.saturating_sub(1));
249        self.cursor.row = new_row.min(self.size.rows.saturating_sub(1));
250
251        self.cursor
252    }

254    /// Erases the last character from the current line buffer
255    pub fn erase_character(&mut self) {
256        if self.current_line.pop().is_some() {
257            if self.cursor.col > 0 {
258                self.cursor.col -= 1;
259            } else if self.cursor.row > 0 {
260                // Wrapped to the previous line
261                self.cursor.row -= 1;
262                self.cursor.row = self.size.rows.saturating_sub(1);
263            }
264        }
265    }

286    /// Completes the current line and adds it to completed lines
287    #[instrument(skip(self))]
288    pub fn complete_line(&mut self) -> SegmentedString {
289        let line = std::mem::take(&mut self.current_line);
290        self.completed_lines.push(line.clone());
291        self.cursor.col = 0;
292        self.cursor.row = (self.cursor.row + 1).min(self.size.rows.saturating_sub(1));
293        line
294    }

331    /// Appends a character to the current line buffer and advances the cursor position.
332    ///
333    /// This method adds a character to the end of the current line and updates the cursor
334    /// position according to the terminal size constraints. The cursor will automatically
335    /// wrap to the next line when reaching the rightmost column, provided there are
336    /// additional rows available.
337    ///
338    /// # Cursor Behavior
339    ///
340    /// The cursor advancement follows these rules:
341    /// - **Normal case**: If the cursor is not at the rightmost column, it advances one position to the right
342    /// - **Line wrap**: If the cursor is at the rightmost column and not on the last row, it wraps to column 0 of the next row
343    /// - **Bottom-right corner**: If the cursor is at the bottom-right corner of the terminal, it remains at the rightmost position
344    ///
345    /// # Terminal Size Handling
346    ///
347    /// The method respects the terminal's dimensions (`self.size.cols` and `self.size.rows`):
348    /// - Characters appended at the right edge trigger line wrapping when possible
349    /// - The cursor never exceeds `cols - 1` or `rows - 1`
350    /// - Uses saturating arithmetic to prevent underflow in boundary calculations
351    ///
352    /// # Arguments
353    ///
354    /// * `c` - The character to append to the current line
355    ///
356    /// # Examples
357    ///
358    /// Basic character appending:
359    ///
360    /// ```rust
361    /// # use termionix_terminal::TerminalBuffer;
362    /// let mut buffer = TerminalBuffer::new_with_size(80, 24);
363    ///
364    /// buffer.append_char('H');
365    /// buffer.append_char('i');
366    ///
367    /// assert_eq!(buffer.current_line_stripped(), "Hi");
368    /// assert_eq!(buffer.cursor_position().col, 2);
369    /// ```
370    ///
371    /// Line wrapping at terminal edge:
372    ///
373    /// ```rust
374    /// # use termionix_terminal::TerminalBuffer;
375    /// let mut buffer = TerminalBuffer::new_with_size(5, 3);
376    ///
377    /// // Fill the first line
378    /// for ch in "Test".chars() {
379    ///     buffer.append_char(ch);
380    /// }
381    ///
382    /// // Cursor should be at rightmost column
383    /// assert_eq!(buffer.cursor_position().col, 4);
384    /// assert_eq!(buffer.cursor_position().row, 0);
385    ///
386    /// // Next character wraps to next line
387    /// buffer.append_char('!');
388    /// assert_eq!(buffer.cursor_position().col, 0);
389    /// assert_eq!(buffer.cursor_position().row, 1);
390    /// ```
391    ///
392    /// Bottom-right corner behavior:
393    ///
394    /// ```rust
395    /// # use termionix_terminal::TerminalBuffer;
396    /// let mut buffer = TerminalBuffer::new_with_size(3, 2);
397    ///
398    /// // Move to bottom-right corner
399    /// buffer.set_cursor_position(2, 1);
400    ///
401    /// // Append character - cursor stays at rightmost position
402    /// buffer.append_char('X');
403    /// assert_eq!(buffer.cursor_position().col, 2);
404    /// assert_eq!(buffer.cursor_position().row, 1);
405    /// ```
406    ///
407    /// Unicode character support:
408    ///
409    /// ```rust
410    /// # use termionix_terminal::TerminalBuffer;
411    /// let mut buffer = TerminalBuffer::new();
412    ///
413    /// buffer.append_char('ðŸ¦€'); // Multi-byte UTF-8 character
414    /// buffer.append_char('æ—¥'); // Japanese character
415    ///
416    /// assert_eq!(buffer.current_line_stripped(), "ðŸ¦€æ—¥");
417    /// assert_eq!(buffer.cursor_position().col, 2);
418    /// ```
419    ///
420    /// # Performance
421    ///
422    /// This is an O(1) operation. The character is appended to the current line's
423    /// internal buffer, and cursor position is updated with simple arithmetic.
424    ///
425    /// # See Also
426    ///
427    /// - [`erase_character()`](TerminalBuffer::erase_character) - Remove the last character and move cursor back
428    /// - [`complete_line()`](TerminalBuffer::complete_line) - Finish the current line and move to the next
429    /// - [`set_cursor_position()`](TerminalBuffer::set_cursor_position) - Manually set cursor position
430    /// - [`move_cursor()`](TerminalBuffer::move_cursor) - Move cursor by relative offset
431    #[instrument(level = "trace", skip(self, c))]
432    pub fn append_char(&mut self, c: char) {
433        // Handle control characters
434        match c {
435            '\x08' => {
436                // Backspace (BS) - erase previous character
437                self.erase_character();
438                return;
439            }
440            '\x07' => {
441                // Bell (BEL) - audible alert (no visual effect on buffer)
442                return;
443            }
444            '\x09' => {
445                // '\t'
446                // Horizontal Tab (HT) - add tab character
447                self.current_line.push_char(c);
448                // Move cursor to next tab stop (typically 8 columns)
449                let tab_width = 8;
450                let next_tab_stop = ((self.cursor.col / tab_width) + 1) * tab_width;
451                self.cursor.col = next_tab_stop.min(self.size.cols.saturating_sub(1));
452                return;
453            }
454            '\x0A' => {
455                // '\n'
456                // Line Feed (LF) - complete the current line and move down
457                self.complete_line();
458                return;
459            }
460            '\x0B' => {
461                // Vertical Tab (VT) - treat like line feed
462                self.complete_line();
463                return;
464            }
465            '\x0C' => {
466                // Form Feed (FF) - clear screen (complete current line, clear all)
467                self.complete_line();
468                self.clear_completed_lines();
469                return;
470            }
471            '\x0D' => {
472                // '\r'
473                // Carriage Return (CR) - move the cursor to the beginning of the line
474                self.cursor.col = 0;
475                return;
476            }
477            '\x7F' => {
478                // Delete (DEL) - same as backspace
479                self.erase_character();
480                return;
481            }
482            // Ignore other C0 control codes (0x00-0x1F) except those handled above
483            '\x00'..='\x1F' => {
484                // Skip other control codes (NUL, SOH, STX, ETX, EOT, ENQ, ACK, etc.)
485                return;
486            }
487            _ => {
488                // Regular printable character - append it
489            }
490        }
491
492        self.current_line.push_char(c);
493
494        // Check if the cursor would exceed terminal width after advancing
495        let max_col = self.size.cols.saturating_sub(1);
496        if self.cursor.col >= max_col {
497            // At or beyond the right edge - wrap to the next line if not at bottom
498            let max_row = self.size.rows.saturating_sub(1);
499            if self.cursor.row < max_row {
500                // Move to the next line
501                self.cursor.col = 0;
502                self.cursor.row += 1;
503            } else {
504                // At the bottom-right corner - stay at rightmost position
505                self.cursor.col = max_col;
506            }
507        } else {
508            // Normal case - just advance cursor
509            self.cursor.col += 1;
510        }
511    }

556    // ============================ Helpers =====================================
557    /// Advance the cursor by one column, clamping to terminal bounds.
558    /// If at the right-most column, do not increment past width - 1.
559    pub fn advance_cursor_by_one(&mut self) {
560        // If the width is zero, stay at column 0
561        if self.size.cols == 0 {
562            self.cursor.col = 0;
563            return;
564        }
565
566        // Only advance if we're not at the right-most column
567        let max_col = self.size.cols.saturating_sub(1);
568        if self.cursor.col < max_col {
569            self.cursor.col += 1;
570        } else {
571            // At right-most column: keep column at max, but if not at the bottom row, allow moving
572            // to the next row on completion of line only.
573            self.cursor.col = max_col;
574        }
575    }
