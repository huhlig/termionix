26/// Terminal commands for output and control
27///
28/// This enum represents all commands that can be sent to a terminal,
29/// including text output, control sequences, and telnet commands.
30#[derive(Clone, Debug, PartialEq, Eq)]
31pub enum TerminalCommand {
32    // Text output variants
33    /// Send a single character
34    Char(char),
35    /// Send text as a string
36    Text(String),
37    /// Send raw bytes
38    Bytes(Vec<u8>),
39
40    // ANSI Sequences
41    /// A C0 or C1 control character.
42    ///
43    /// These are non-printable control codes in the ranges:
44    /// - C0: 0x00-0x1F (excluding ESC at 0x1B)
45    /// - DEL: 0x7F
46    /// - C1: 0x80-0x9F
47    ///
48    /// Common examples include NULL (0x00), Bell (0x07), Backspace (0x08),
49    /// Tab (0x09), Line Feed (0x0A), and Carriage Return (0x0D).
50    Control(AnsiControlCode),
51
52    /// A standalone ESC character (0x1B) that is not part of a recognized sequence.
53    ///
54    /// This occurs when an ESC character is followed by a byte that doesn't
55    /// initiate a known ANSI escape sequence. The ESC was not consumed as part
56    /// of a control sequence.
57    AnsiEscape,
58
59    /// Control Sequence Introducer - a general CSI escape sequence.
60    ///
61    /// Format: `ESC [ <params> <final_byte>`
62    ///
63    /// CSI sequences are used for cursor movement, screen manipulation, and other
64    /// terminal control operations. The final byte (0x40-0x7E) determines the
65    /// specific command. Common examples:
66    /// - `ESC[H` - Cursor Home
67    /// - `ESC[2J` - Clear Screen
68    /// - `ESC[10;20H` - Move cursor to row 10, column 20
69    ///
70    /// Note: SGR sequences (ending with 'm') are parsed separately and returned
71    /// as the `SGR` variant instead.
72    AnsiCSI(AnsiControlSequenceIntroducer),
73
74    /// Select Graphic Rendition - a specialized CSI sequence for text styling.
75    ///
76    /// Format: `ESC [ <params> m`
77    ///
78    /// SGR sequences control text appearance including colors, bold, italic,
79    /// underline, and other visual attributes. This is a specialized form of
80    /// CSI sequence that is parsed into a `Style` object for convenience.
81    ///
82    /// Examples:
83    /// - `ESC[0m` - Reset all attributes
84    /// - `ESC[1m` - Bold
85    /// - `ESC[31m` - Red foreground
86    /// - `ESC[1;31;42m` - Bold red text on a green background
87    AnsiSGR(AnsiSelectGraphicRendition),
88
89    /// Operating System Command - a sequence for terminal-specific operations.
90    ///
91    /// Format: `ESC ] <params> ST` or `ESC ] <params> BEL`
92    ///
93    /// OSC sequences communicate with the terminal emulator to perform operations
94    /// like setting the window title, changing color palettes, or other OS-level
95    /// terminal features. The sequence is terminated by either ST (String Terminator,
96    /// ESC \) or BEL (0x07).
97    ///
98    /// The raw bytes (excluding the terminator) are returned for interpretation
99    /// by the application.
100    AnsiOSC(AnsiOperatingSystemCommand),
101
102    /// Device Control String - a sequence for device-specific control.
103    ///
104    /// Format: `ESC P <params> ST`
105    ///
106    /// DCS sequences are used to send device-specific control strings to the
107    /// terminal. They are terminated by ST (ESC \). The contents are device-
108    /// dependent and returned as raw bytes.
109    AnsiDCS(AnsiDeviceControlString),
110
111    /// Start of String - a legacy control sequence.
112    ///
113    /// Format: `ESC X <data> ST`
114    ///
115    /// SOS is a rarely used control function from ISO 6429. It marks the start
116    /// of a control string that is terminated by ST (ESC \). The contents are
117    /// returned as raw bytes.
118    AnsiSOS(AnsiStartOfString),
119
120    /// String Terminator - marks the end of a string control sequence.
121    ///
122    /// Format: `ESC \`
123    ///
124    /// ST is used to terminate string-type control sequences (OSC, DCS, SOS, PM, APC).
125    /// When encountered outside of a string sequence context, it's returned as a
126    /// standalone result with empty data.
127    AnsiST,
128
129    /// Privacy Message - a control sequence for private data.
130    ///
131    /// Format: `ESC ^ <data> ST`
132    ///
133    /// PM is a control function from ISO 6429 used to delimit privacy messages.
134    /// The sequence is terminated by ST (ESC \) and the contents are returned
135    /// as raw bytes.
136    AnsiPM(AnsiPrivacyMessage),
137
138    /// Application Program Command - a control sequence for application-specific commands.
139    ///
140    /// Format: `ESC _ <data> ST`
141    ///
142    /// APC sequences allow applications to send custom commands through the
143    /// terminal. The sequence is terminated by ST (ESC \), and the contents are
144    /// returned as raw bytes for application-specific interpretation.
145    AnsiAPC(AnsiApplicationProgramCommand),
146
147    // Telnet Commands
148    /// No Operation - Does nothing but sends an IAC command.
149    ///
150    /// Format: `IAC NOP` (0xFF 0xF1)
151    ///
152    /// This command is often used as a keep-alive or to test the connection.
153    NoOperation,
154
155    /// End of urgent Data Stream - Marks the end of urgent data.
156    ///
157    /// Format: `IAC DM` (0xFF 0xF2)
158    ///
159    /// Used to synchronize urgent and normal data streams in the connection.
160    DataMark,
161
162    /// Break - Operator pressed the Break key or the Attention key.
163    ///
164    /// Format: `IAC BRK` (0xFF 0xF3)
165    ///
166    /// Sends an interrupt signal to the remote system, typically used to interrupt
167    /// a running process on the server.
168    Break,
169
170    /// Interrupt Process - Request immediate interrupt of the current process.
171    ///
172    /// Format: `IAC IP` (0xFF 0xF4)
173    ///
174    /// Sends a stronger interrupt signal than Break, typically mapped to SIGINT in Unix-like systems.
175    InterruptProcess,
176
177    /// Cancel Output - Request that the remote system cancel output to the client.
178    ///
179    /// Format: `IAC AO` (0xFF 0xF5)
180    ///
181    /// Used when the client no longer wants to receive output from the remote process.
182    AbortOutput,
183
184    /// Are You There - Request acknowledgment from the remote system.
185    ///
186    /// Format: `IAC AYT` (0xFF 0xF6)
187    ///
188    /// Used to test if the connection is still alive or if the remote host is responding.
189    AreYouThere,
190
191    /// Erase Character - Request that the operator erase the previous character.
192    ///
193    /// Format: `IAC EC` (0xFF 0xF7)
194    ///
195    /// Equivalent to sending a backspace or delete request to the remote system.
196    EraseCharacter,
197
198    /// Erase Line - Request that the operator erase the previous line.
199    ///
200    /// Format: `IAC EL` (0xFF 0xF8)
201    ///
202    /// Requests the remote system to clear the current input line.
203    EraseLine,
204
205    /// Go Ahead - End of input for half-duplex connections.
206    ///
207    /// Format: `IAC GA` (0xFF 0xF9)
208    ///
209    /// Used in half-duplex mode to indicate that the sender has finished transmitting
210    /// and the receiver may now send data. Rarely used in modern systems.
211    GoAhead,
212
213    /// End of Record - Marks the end of a prompt.
214    ///
215    /// Format: `IAC EOR` (0xFF 0xEF)
216    ///
217    /// Used by MUD servers to mark the end of a prompt. A prompt is considered
218    /// any line that does not end with \r\n. This allows clients to distinguish
219    /// between regular output and prompts that require user input.
220    EndOfRecord,
221
222    // Telnet Subnegotation Messages
223    /// Generic Mud Communication Protocol
224    GMCP(GmcpMessage),
225    /// Mud Server Data Protocol
226    MSDP(MudServerData),
227    /// Mud Server Status Protocol
228    MSSP(MudServerStatus),
229}

310impl From<AnsiSequence> for TerminalCommand {
311    fn from(seq: AnsiSequence) -> Self {
312        match seq {
313            AnsiSequence::Character(c) => TerminalCommand::Char(c),
314            AnsiSequence::Unicode(c) => TerminalCommand::Char(c),
315            AnsiSequence::AnsiControlCode(code) => TerminalCommand::Control(code),
316            AnsiSequence::AnsiEscape => TerminalCommand::AnsiEscape,
317            AnsiSequence::AnsiCSI(csi) => TerminalCommand::AnsiCSI(csi),
318            AnsiSequence::AnsiSGR(sgr) => TerminalCommand::AnsiSGR(sgr),
319            AnsiSequence::AnsiOSC(osc) => TerminalCommand::AnsiOSC(osc),
320            AnsiSequence::AnsiDCS(dcs) => TerminalCommand::AnsiDCS(dcs),
321            AnsiSequence::AnsiSOS(sos) => TerminalCommand::AnsiSOS(sos),
322            AnsiSequence::AnsiST => TerminalCommand::AnsiST,
323            AnsiSequence::AnsiPM(pm) => TerminalCommand::AnsiPM(pm),
324            AnsiSequence::AnsiAPC(apc) => TerminalCommand::AnsiAPC(apc),
325            AnsiSequence::TelnetCommand(cmd) => {
326                // Convert TelnetCommand to TerminalCommand
327                use termionix_ansicodec::TelnetCommand as TC;
328                match cmd {
329                    TC::NoOperation => TerminalCommand::NoOperation,
330                    TC::DataMark => TerminalCommand::DataMark,
331                    TC::Break => TerminalCommand::Break,
332                    TC::InterruptProcess => TerminalCommand::InterruptProcess,
333                    TC::AbortOutput => TerminalCommand::AbortOutput,
334                    TC::AreYouThere => TerminalCommand::AreYouThere,
335                    TC::EraseCharacter => TerminalCommand::EraseCharacter,
336                    TC::EraseLine => TerminalCommand::EraseLine,
337                    TC::GoAhead => TerminalCommand::GoAhead,
338                    TC::EndOfRecord => TerminalCommand::EndOfRecord,
339                    // For subnegotiation commands, we'd need to handle them specially
340                    // For now, just use NoOperation as a placeholder
341                    _ => TerminalCommand::NoOperation,
342                }
343            }
344        }
345    }
346}
