17use super::TelnetOption;
18use crate::args::TelnetArgument;
19
20///
21/// `TelnetFrame` is a single atomic Frame of data.
22///
23#[derive(Clone, Debug, PartialEq)]
24pub enum TelnetFrame {
25    /// Telnet Data Byte
26    Data(u8),
27    /// No Operation
28    NoOperation,
29    /// End of urgent Data Stream
30    DataMark,
31    /// Operator pressed the Break key or the Attention key.
32    Break,
33    /// Interrupt current process.
34    InterruptProcess,
35    /// Cancel output from the current process.
36    AbortOutput,
37    /// Request acknowledgment.
38    AreYouThere,
39    /// Request that the operator erase the previous character.
40    EraseCharacter,
41    /// Request that the operator erase the previous line.
42    EraseLine,
43    /// End of input for half-duplex connections.
44    GoAhead,
45    /// End of Record - marks the end of a prompt
46    EndOfRecord,
47    /// Telnet Do Option
48    Do(TelnetOption),
49    /// Telnet Dont Option
50    Dont(TelnetOption),
51    /// Telnet Will Option
52    Will(TelnetOption),
53    /// Telnet Wont Option
54    Wont(TelnetOption),
55    /// Subnegotiation Payload
56    Subnegotiate(TelnetArgument),
57}
