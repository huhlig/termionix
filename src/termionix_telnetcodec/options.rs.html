390/// A structure representing the configuration and state of Telnet options.
391///
392/// The `TelnetOptions` struct is designed to manage the options available within
393/// a Telnet sidechannel implementation. It tracks the support state of each option
394/// as well as the current negotiation state for each option.
395///
396/// # Fields
397///
398/// * `config` - An array of `SupportState` values with a fixed size of 255, representing
399///              the support state of each Telnet option. Each index corresponds to the
400///              associated Telnet option code, determining whether the option is supported
401///              and in what manner.
402///
403/// * `state` - An array of `OptionState` values with a fixed size of 255, representing
404///             the negotiation state of each Telnet option. Each index corresponds to the
405///             associated Telnet option code, tracking the current state of the option
406///             in terms of enabling, disabling, and negotiation progress.
407///
408/// # Notes
409///
410/// The Telnet sidechannel uses option codes ranging from 0 to 254, which makes the size
411/// of both arrays precisely 255 to represent all potential options. The fields ensure
412/// the ability to handle and manage all standard Telnet options.

576    /// Handles a received Telnet frame and processes it based on its type.

581    /// - `TelnetFrame::Dont`: Calls `recv_dont` to process the frame.
582    /// - `TelnetFrame::Will`: Calls `recv_will` to process the frame.
583    /// - `TelnetFrame::Wont`: Calls `recv_wont` to process the frame.
584    /// - Other frame types result in a `TerminalError::NegotationError`.
585    ///
586    /// # Arguments
587    ///
588    /// * `frame` - A `TelnetFrame` that represents the received Telnet command or negotiation option.
589    ///
590    /// # Returns
591    ///
592    /// Returns a `TerminalResult<Option<TelnetFrame>>`, which can be one of:
593    /// - `Ok(Some(TelnetFrame))`: Indicates a successful processing of the received frame, possibly
594    ///   including a response frame to be sent.
595    /// - `Ok(None)`: Indicates a successful processing of the received frame without requiring a response.
596    /// - `Err(TerminalError)`: Indicates a failure during processing, such as an unexpected frame type.
597    ///
598    /// # Errors
599    ///
600    /// This function returns `Err(TerminalError::NegotationError)` if the received `frame`
601    /// is of an unsupported or invalid type.
602    ///
603    pub fn handle_received(
604        &mut self,
605        frame: TelnetFrame,
606    ) -> TelnetCodecResult<Option<TelnetFrame>> {
607        match frame {
608            TelnetFrame::Do(option) => Ok(self.recv_do(option)),
609            TelnetFrame::Dont(option) => Ok(self.recv_dont(option)),
610            TelnetFrame::Will(option) => Ok(self.recv_will(option)),
611            TelnetFrame::Wont(option) => Ok(self.recv_wont(option)),
612            _ => Err(TelnetCodecError::NegotiationError {
613                reason: "Unsupported frame type".into(),
614                frame_type: Some(format!("{:?}", frame)),
615            }),
616        }
617    }

621    /// Request that *we* enable the option (i.e. we want to send WILL).
622    /// Returns frames (commands) you should send on the wire as a result.
623    fn request_will(&mut self, option: TelnetOption) -> Option<TelnetFrame> {
624        // First check if we support providing this option
625        if !self.config[option.to_u8() as usize].local {
626            return None; // Don't try to enable unsupported options
627        }
628        match self.state[option.to_u8() as usize].local {
629            QState::Yes | QState::WantYes | QState::WantYesOpposite => {
630                // already enabled or in-progress to enable
631                None
632            }
633            QState::No => {
634                // Start negotiation: send WILL -> state WANTYES
635                self.state[option.to_u8() as usize].local = QState::WantYes;
636                Some(TelnetFrame::Will(option))
637            }
638            QState::WantNo => {
639                // collision: we were trying to disable, but now ask enable -> WantYesOpposite
640                self.state[option.to_u8() as usize].local = QState::WantYesOpposite;
641                Some(TelnetFrame::Will(option))
642            }
643            QState::WantNoOpposite => {
644                // will -> wantyes (opposite canceled)
645                self.state[option.to_u8() as usize].local = QState::WantYes;
646                Some(TelnetFrame::Will(option))
647            }
648        }
649    }

656                None

672        }

718
719    /// Process an incoming WILL from remote (they say "I will do option").
720    /// Returns frames to send in response (if any).
721    fn recv_will(&mut self, option: TelnetOption) -> Option<TelnetFrame> {
722        let option_idx = option.to_u8() as usize;
723        // Handle out-of-bounds option (e.g., Unknown option with value 255)
724        if option_idx >= self.config.len() {
725            return Some(TelnetFrame::Dont(option));
726        }
727        // First, check if we support providing this option
728        if !self.config[option_idx].remote {
729            // If we don't support it, reject with `DONT`
730            return Some(TelnetFrame::Dont(option));
731        }
732        match self.state[option_idx].remote {
733            QState::No => {
734                // remote offers WILL -> if we accept, send DO and move to YES
735                // For a generic engine we accept by default; caller can override by sending DONT.
736                // Here we accept. If you want to implement policy, change this branch.
737                self.state[option_idx].remote = QState::Yes;
738                Some(TelnetFrame::Do(option))
739            }
740            QState::Yes => {
741                // already yes -> no response
742                None
743            }
744            QState::WantNo => {
745                // peer is contradicting our previous DONT: move to No (or remain?) RFC1143:
746                // WANTNO + WILL => WANTNO-OPPOSITE -> send DONT
747                self.state[option_idx].remote = QState::WantNoOpposite;
748                Some(TelnetFrame::Dont(option))
749            }
750            QState::WantNoOpposite => {
751                // collision resolved -> YES
752                self.state[option_idx].remote = QState::Yes;
753                None // no further response
754            }
755            QState::WantYes => {
756                // we asked for it, and peer confirms -> YES
757                self.state[option_idx].remote = QState::Yes;
758                None
759            }
760            QState::WantYesOpposite => {
761                // double negotiation: move to YES
762                self.state[option_idx].remote = QState::Yes;
763                None
764            }
765        }
766    }

768    /// Process incoming WONT from remote.
769    fn recv_wont(&mut self, option: TelnetOption) -> Option<TelnetFrame> {
770        let option_idx = option.to_u8() as usize;
771        // Handle out-of-bounds option (e.g., Unknown option with value 255)
772        if option_idx >= self.state.len() {
773            return None; // No response needed for unknown options
774        }
775        match self.state[option_idx].remote {
776            QState::No => None, // already no
777            QState::Yes => {

779                // if we expected it, nothing to send; RFC1143: no immediate reply
780                None
781            }
782            QState::WantNo => {
783                self.state[option_idx].remote = QState::No;
784                None
785            }
786            QState::WantNoOpposite => {
787                // remote confirmed refusal -> NO
788                self.state[option_idx].remote = QState::No;
789                None
790            }
791            QState::WantYes => {
792                // requested YES, but peer refuses -> NO and maybe clear want
793                self.state[option_idx].remote = QState::No;
794                None
795            }
796            QState::WantYesOpposite => {
797                self.state[option_idx].remote = QState::No;
798                None
799            }
800        }
801    }

820            }
821            QState::Yes => None,
822            QState::WantNo => {
823                self.state[option_idx].local = QState::WantNoOpposite;
824                Some(TelnetFrame::Will(option))
825            }
826            QState::WantNoOpposite => {
827                self.state[option_idx].local = QState::Yes;
828                None
829            }
830            QState::WantYes => {
831                self.state[option_idx].local = QState::Yes;
832                None
833            }
834            QState::WantYesOpposite => {
835                self.state[option_idx].local = QState::Yes;
836                None
837            }
838        }
839    }

897/// Represents the perspective of a Telnet option in a client-server negotiation.
898///
899/// In the Telnet sidechannel, option negotiation involves two independent paths:
900/// one for local options (what the local side wants to do) and one for remote options
901/// (what the remote side wants to do). `OptionSide` disambiguates between these two
902/// perspectives when managing option state.
903///
904/// # Variants
905///
906/// ## `Local`
907///
908/// Represents the local side of the connection. This is the perspective of the endpoint
909/// that is currently executing this code.
910///
911/// When applied to option negotiation:
912/// - Used when *we* want to enable or disable an option
913/// - We send `WILL` (agreement to perform) or `WONT` (refusal to perform) commands
914/// - The remote side responds with `DO` (request us to perform) or `DONT` (request us not to perform)
915///
916/// # Examples
917///
918/// Requesting that *we* start performing an option:
919/// ```text
920/// Local: WILL <option>    Remote
921/// Remote: DO <option>     Local
922/// ```
923///
924/// ## `Remote`
925///
926/// Represents the remote side of the connection. This is the perspective of the peer
927/// endpoint that we are communicating with.
928///
929/// When applied to option negotiation:
930/// - Used when *we* want the remote side to enable or disable an option
931/// - We send `DO` (request them to perform) or `DONT` (request them not to perform) commands
932/// - The remote side responds with `WILL` (agreement to perform) or `WONT` (refusal to perform)
933///
934/// # Examples
935///
936/// Requesting that the *remote side* start performing an option:
937/// ```text
938/// Local: DO <option>      Remote
939/// Remote: WILL <option>   Local
940/// ```
941///
942/// # Usage in Option State Machine
943///
944/// Each Telnet option maintains two independent state machines (using the Q-method defined in RFC 1143):
945/// one for the local side and one for the remote side. `OptionSide` is used to determine which
946/// state machine path to follow:
947///
948/// - **Local path**: Tracks whether *we* are performing an option
949///   - Initial commands: `WILL` / `WONT`
950///   - Expected responses: `DO` / `DONT`
951///   - Final states: enabled (performing) or disabled (not performing)
952///
953/// - **Remote path**: Tracks whether the *remote side* is performing an option
954///   - Initial commands: `DO` / `DONT`
955///   - Expected responses: `WILL` / `WONT`
956///   - Final states: enabled (remote is performing) or disabled (remote not performing)
957///
958/// # See Also
959///
960/// - [`TelnetOption`]: The specific Telnet option being negotiated
961/// - [`TelnetOptions`]: Manages both local and remote state for all options
962/// - [`QState`]: The negotiation state machine states
963///
964#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
965pub enum TelnetSide {
966    /// The local side of the Telnet connection (what we want to do)
967    Local,
968    /// The remote side of the Telnet connection (what the peer wants to do)
969    Remote,
970}

972impl std::fmt::Display for TelnetSide {
973    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
974        match self {
975            TelnetSide::Local => write!(f, "Local"),
976            TelnetSide::Remote => write!(f, "Remote"),
977        }
978    }
979}

1004/// Represents the state of a negotiation process with possible outcomes and desires.

1032    ///
1033    #[default]
1034    No,
1035    ///
1036    WantNo,
1037    ///
1038    WantNoOpposite,
1039    ///
1040    Yes,
1041    ///
1042    WantYes,
