17//! MudServerStatus Option
18//!
19//! https://tools.ietf.org/html/rfc8549#section-3.1.2
20//!
21//! MSSP is a subnegotiation of the MSSP option.
22//!
23//! The MSSP subnegotiation is used to send information about the Mud to the
24//! client. The information is sent in a series of key-value pairs.
25//!
26//!
27use crate::{TelnetCodecResult, consts};
28use byteorder::WriteBytesExt;
29use bytes::BufMut;
30use std::collections::HashMap;
31
32/// Mud Server Status Protocol handler for TELNET negotiation.
33///
34/// This struct manages MUD server information transmission according to the
35/// [MSSP Protocol](https://tintin.sourceforge.io/protocols/mssp/).
36///
37/// MSSP is a subnegotiation of the TELNET MSSP option that allows servers to send
38/// structured information about the MUD to clients in key-value pairs. Each key can
39/// have multiple associated values.
40///
41/// # Examples
42///
43/// ```
44/// use termionix_telnetcodec::mssp::MudServerStatus;
45///
46/// let mut status = MudServerStatus::new();
47/// // Add server information...
48/// ```
49#[derive(Clone, Debug, Eq, PartialEq)]
50pub struct MudServerStatus(HashMap<String, Vec<String>>);
51
52impl MudServerStatus {
53    /// Creates a new, empty `MudServerStatus` instance.
54    ///
55    /// Initializes an empty MSSP status with no key-value pairs.
56    ///
57    /// # Returns
58    ///
59    /// A new `MudServerStatus` with no entries. The internal HashMap is empty.
60    ///
61    /// # Examples
62    ///
63    /// ```
64    /// use telnetcodec::mssp::MudServerStatus;
65    ///
66    /// let status = MudServerStatus::new();
67    /// assert_eq!(status.len(), 0);
68    /// ```
69    pub fn new() -> MudServerStatus {
70        MudServerStatus(HashMap::new())
71    }
72
73    /// Returns the encoded length of the `MudServerStatus` in bytes.
74    ///
75    /// Calculates the total number of bytes that will be needed to encode this
76    /// `MudServerStatus` according to the MSSP sidechannel. This accounts for all
77    /// keys, values, and their separator markers but does not include the effect
78    /// of character filtering that occurs during actual encoding.
79    ///
80    /// # Returns
81    ///
82    /// The total encoded length in bytes. This is the sum of:
83    /// - 1 byte for each key's VAR marker
84    /// - The length of each key string
85    /// - 1 byte for each value's VAL marker
86    /// - The length of each value string
87    ///
88    /// # Examples
89    ///
90    /// ```
91    /// use termionix_telnetcodec::mssp::MudServerStatus;
92    ///
93    /// let status = MudServerStatus::new();
94    /// assert_eq!(status.len(), 0);
95    /// ```
96    pub fn len(&self) -> usize {
97        let mut length = 0;
98        for (key, values) in &self.0 {
99            length += 1;
100            length += key.len();
101            for value in values {
102                length += 1;
103                length += value.len();
104            }
105        }
106        length
107    }
108
109    /// Encodes `MudServerStatus` to a `BufMut` buffer.
110    ///
111    /// Serializes the MSSP status information into the provided mutable buffer.
112    /// Invalid characters (NUL, IAC, VAR, VAL) are filtered out from keys and values
113    /// during encoding to ensure sidechannel compliance.
114    ///
115    /// # Arguments
116    ///
117    /// * `dst` - A mutable buffer that implements `BufMut` where the encoded data
118    ///          will be written
119    ///
120    /// # Returns
121    ///
122    /// * `Ok(usize)` - The number of bytes written to the buffer
123    /// * `Err(CodecResult)` - If an I/O error occurs during encoding
124    ///
125    /// # Examples
126    ///
127    /// ```
128    /// use termionix_telnetcodec::mssp::MudServerStatus;
129    /// use bytes::BytesMut;
130    ///
131    /// let status = MudServerStatus::new();
132    /// let mut buffer = BytesMut::new();
133    /// match status.encode(&mut buffer) {
134    ///     Ok(bytes_written) => println!("Encoded {} bytes", bytes_written),
135    ///     Err(e) => eprintln!("Encoding error: {:?}", e),
136    /// }
137    /// ```
138    pub fn encode<T: BufMut>(&self, dst: &mut T) -> TelnetCodecResult<usize> {
139        Ok(self.write(&mut dst.writer())?)
140    }
141
142    /// Writes the encoded `MudServerStatus` to a byte writer.
143    ///
144    /// Low-level method that serializes the MSSP status to an implementer of
145    /// `std::io::Write`. Keys and values are filtered to remove any characters
146    /// that are invalid in the MSSP sidechannel (NUL, IAC, VAR, VAL).
147    ///
148    /// The encoding format uses:
149    /// - VAR marker (0x01) to prefix each key
150    /// - VAL marker (0x02) to prefix each value
151    ///
152    /// # Arguments
153    ///
154    /// * `writer` - A mutable reference to a type implementing `std::io::Write`
155    ///             where the encoded bytes will be written
156    ///
157    /// # Returns
158    ///
159    /// * `Ok(usize)` - The total number of bytes written
160    /// * `Err(std::io::Error)` - If a write error occurs
161    ///
162    /// # Character Filtering
163    ///
164    /// The following characters are removed from keys and values:
165    /// - NUL (0x00) - Null terminator
166    /// - IAC (0xFF) - TELNET Interpret As Command
167    /// - VAR (0x01) - MSSP variable marker
168    /// - VAL (0x02) - MSSP value marker
169    ///
170    /// # Examples
171    ///
172    /// ```
173    /// use termionix_telnetcodec::mssp::MudServerStatus;
174    /// use std::io::Cursor;
175    ///
176    /// let status = MudServerStatus::new();
177    /// let mut buffer = Vec::new();
178    /// match status.write(&mut buffer) {
179    ///     Ok(bytes_written) => println!("Wrote {} bytes", bytes_written),
180    ///     Err(e) => eprintln!("Write error: {}", e),
181    /// }
182    /// ```
183    pub fn write<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<usize> {
184        let mut len = 0;
185        for (key, values) in &self.0 {
186            writer.write_u8(consts::option::mssp::VAR)?;
187            len += 1 + writer.write(
188                key.chars()
189                    .filter(|ch| {
190                        *ch != consts::NUL as char
191                            && *ch != consts::IAC as char
192                            && *ch != consts::option::mssp::VAR as char
193                            && *ch != consts::option::mssp::VAL as char
194                    })
195                    .collect::<String>()
196                    .as_bytes(),
197            )?;
198            for value in values {
199                writer.write_u8(consts::option::mssp::VAL)?;
200                len += 1 + writer.write(
201                    value
202                        .chars()
203                        .filter(|ch| {
204                            *ch != consts::NUL as char
205                                && *ch != consts::IAC as char
206                                && *ch != consts::option::mssp::VAR as char
207                                && *ch != consts::option::mssp::VAL as char
208                        })
209                        .collect::<String>()
210                        .as_bytes(),
211                )?;
212            }
213        }
214        Ok(len)
215    }
216}

219    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
220        std::fmt::Debug::fmt(self, f)
221    }
