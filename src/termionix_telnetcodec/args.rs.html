36///
37/// Telnet Subnegotiation Argument
38///
39#[derive(Clone, Debug, PartialEq, Eq)]
40pub enum TelnetArgument {
41    /// A subnegotiation for the window size, where the first value is the width
42    /// and the second value is the height. The values are in characters.
43    NAWSWindowSize(WindowSize),
44    /// Indicates an intent to begin CHARSET subnegotiation. This can only be
45    /// sent after receiving a DO CHARSET after sending a WILL CHARSET (in any
46    /// order).
47    CharsetRequest(Vec<BytesMut>),
48    /// Indicates that the receiver has accepted the charset request.
49    CharsetAccepted(BytesMut),
50    /// Indicates that the receiver acknowledges the charset request but will
51    /// not use any of the requested characters.
52    CharsetRejected,
53    /// Indicates that the receiver acknowledges a TTABLE-IS message but is
54    /// unable to handle it. This will terminate subnegotiation.
55    CharsetTTableRejected,
56    /// GMCP (Generic Mud Communication Protocol) message.
57    /// Contains a package name and optional JSON data payload.
58    GMCP(GmcpMessage),
59    /// Mud Server Data message.
60    MudServerData(MudServerData),
61    /// Mud Server Status message.
62    MudServerStatus(MudServerStatus),
63    /// Terminal type subnegotiation.
64    TerminalType(String),
65    /// A subnegotiation for an unknown option.
66    Unknown(TelnetOption, BytesMut),
67}
68
69impl TelnetArgument {
70    /// Returns the encoded byte length of this `TelnetArgument`.
71    ///
72    /// Calculates the total number of bytes that will be produced when this argument
73    /// is encoded to its wire format representation. This is useful for pre-allocating
74    /// buffers before encoding.
75    ///
76    /// # Returns
77    ///
78    /// The number of bytes in the encoded form.
79    ///
80    /// # Behavior by Variant
81    ///
82    /// - `NAWSWindowSize(inner)` - Returns the encoded length of the window size data
83    /// - `Unknown(option, payload)` - Returns the length of the payload bytes
84    /// - Other variants - Currently unimplemented
85    ///
86    /// # Performance
87    ///
88    /// This is an O(1) operation for most variants, though some may require
89    /// data length inspection.
90    ///
91    /// # Examples
92    ///
93    /// ```
94    /// use termionix_telnetcodec::TelnetArgument;
95    /// use termionix_telnetcodec::naws::WindowSize;
96    ///
97    /// let arg = TelnetArgument::NAWSWindowSize(WindowSize::new(80, 24));
98    /// let size = arg.len();
99    /// ```
100    pub fn len(&self) -> usize {
101        match self {
102            TelnetArgument::NAWSWindowSize(inner) => inner.len(),
103            TelnetArgument::GMCP(inner) => inner.len(),
104            TelnetArgument::Unknown(_option, inner) => inner.len(),
105            _ => unimplemented!(),
106        }
107    }
108    /// Encodes this `TelnetArgument` to a `BufMut` buffer.
109    ///
110    /// Writes the argument's byte representation to a mutable buffer, automatically
111    /// advancing the buffer's write position. This is the preferred method for encoding
112    /// into buffers that implement `BufMut`.
113    ///
114    /// # Arguments
115    ///
116    /// * `dst` - A mutable buffer implementing `BufMut` (e.g., `BytesMut`).
117    ///   The buffer is advanced by the bytes written.
118    ///
119    /// # Returns
120    ///
121    /// Returns `Ok(bytes_written)` on success, containing the exact number of bytes
122    /// that were written to the buffer. Returns a `CodecResult` error if encoding fails.
123    ///
124    /// # Performance
125    ///
126    /// This is an O(n) operation where n is the length of the encoded argument.
127    /// It performs a single pass through the data, writing directly to the buffer
128    /// without intermediate allocations.
129    ///
130    /// # Buffer Behavior
131    ///
132    /// - The buffer is only written to; existing content is preserved
133    /// - The buffer's write position is advanced automatically
134    /// - No bounds checking is performed; the buffer must have sufficient capacity
135    /// - For `BytesMut`, automatic growth occurs if needed
136    ///
137    /// # Examples
138    ///
139    /// ```
140    /// use bytes::BytesMut;
141    /// use termionix_telnetcodec::naws::WindowSize;
142    /// use termionix_telnetcodec::TelnetArgument;
143    ///
144    /// let arg = TelnetArgument::NAWSWindowSize(WindowSize::new(80, 24));
145    /// let mut buffer = BytesMut::new();
146    /// let written = arg.encode(&mut buffer)?;
147    /// println!("Wrote {} bytes", written);
148    /// # Ok::<(), Box<dyn std::error::Error>>(())
149    /// ```
150    pub fn encode<T: BufMut>(&self, dst: &mut T) -> TelnetCodecResult<usize> {
151        Ok(self.write(&mut dst.writer())?)
152    }
153
154    /// Writes this `TelnetArgument` to a `std::io::Write` writer.
155    ///
156    /// Performs the actual encoding of the argument into its byte representation
157    /// and writes the bytes to the provided writer. This is the low-level method
158    /// that `encode()` delegates to internally.
159    ///
160    /// # Arguments
161    ///
162    /// * `writer` - A mutable writer implementing `std::io::Write` (e.g., `Vec<u8>`,
163    ///   `File`, socket, etc.). The writer's internal position is advanced by the
164    ///   bytes written.
165    ///
166    /// # Returns
167    ///
168    /// Returns `Ok(bytes_written)` on success, containing the exact number of bytes written.
169    /// Returns `std::io::Error` if the writer fails (e.g., disk full, broken pipe).
170    ///
171    /// # Encoding Behavior by Variant
172    ///
173    /// - `NAWSWindowSize(inner)` - Delegates to the window size's `write()` method
174    /// - `Unknown(option, payload)` - Writes the raw payload bytes as-is
175    /// - Other variants - Will panic (currently unimplemented)
176    ///
177    /// # Examples
178    ///
179    /// ```
180    /// use bytes::BytesMut;
181    /// use termionix_telnetcodec::{TelnetArgument, TelnetOption};
182    ///
183    /// let arg = TelnetArgument::Unknown(
184    ///     TelnetOption::NAWS,
185    ///     BytesMut::from(&b"\x00\x50\x00\x18"[..])
186    /// );
187    /// let mut output = Vec::new();
188    /// let written = arg.write(&mut output)?;
189    /// assert_eq!(written, 4);
190    /// # Ok::<(), std::io::Error>(())
191    /// ```
192    pub fn write<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<usize> {
193        match self {
194            TelnetArgument::NAWSWindowSize(inner) => inner.write(writer),
195            TelnetArgument::GMCP(inner) => inner.write(writer),
196            TelnetArgument::Unknown(_option, payload) => {
197                // Write payload with IAC escaping
198                let mut written = 0;
199                for &byte in payload.iter() {
200                    if byte == 0xFF {
201                        // IAC byte must be escaped as IAC IAC
202                        writer.write_all(&[0xFF, 0xFF])?;
203                        written += 2;
204                    } else {
205                        writer.write_all(&[byte])?;
206                        written += 1;
207                    }
208                }
209                Ok(written)
210            }
211            _ => unimplemented!(),
212        }
213    }
214
215    /// Returns the `TelnetOption` associated with this argument.
216    ///
217    /// Identifies which Telnet option this argument is for, enabling proper routing
218    /// and handling of subnegotiation data during sidechannel processing.
219    ///
220    /// # Returns
221    ///
222    /// A `TelnetOption` enum value corresponding to the argument's option type.
223    ///
224    /// # Behavior by Variant
225    ///
226    /// - `NAWSWindowSize(_)` → Returns `TelnetOption::NAWS`
227    /// - `CharsetRequest(_)` → Returns `TelnetOption::Charset`
228    /// - `CharsetAccepted(_)` → Returns `TelnetOption::Charset`
229    /// - `CharsetRejected` → Returns `TelnetOption::Charset`
230    /// - `CharsetTTableRejected` → Returns `TelnetOption::Charset`
231    /// - `Unknown(option, _)` → Returns the contained `option`
232    ///
233    /// # Performance
234    ///
235    /// This is an O(1) operation that simply matches and returns a value.
236    ///
237    /// # Examples
238    ///
239    /// ```
240    /// use termionix_telnetcodec::{TelnetArgument, TelnetOption};
241    /// use termionix_telnetcodec::naws::WindowSize;
242    ///
243    /// let arg = TelnetArgument::NAWSWindowSize(WindowSize::new(80, 24));
244    /// assert_eq!(arg.option(), TelnetOption::NAWS);
245    ///
246    /// let arg = TelnetArgument::CharsetRequest(vec![]);
247    /// assert_eq!(arg.option(), TelnetOption::Charset);
248    /// ```
249    ///
250    /// # Use Cases
251    ///
252    /// - **Option routing**: Direct subnegotiation data to the correct handler
253    /// - **Protocol validation**: Verify option consistency in negotiation sequences
254    /// - **Option tracking**: Maintain state of which options are active
255    pub fn option(&self) -> TelnetOption {
256        match self {
257            TelnetArgument::NAWSWindowSize(_) => TelnetOption::NAWS,
258            TelnetArgument::CharsetRequest(_) => TelnetOption::Charset,
259            TelnetArgument::CharsetAccepted(_) => TelnetOption::Charset,
260            TelnetArgument::CharsetRejected => TelnetOption::Charset,
261            TelnetArgument::CharsetTTableRejected => TelnetOption::Charset,
262            TelnetArgument::GMCP(_) => TelnetOption::GMCP,
263            TelnetArgument::MudServerData(_) => TelnetOption::MSDP,
264            TelnetArgument::MudServerStatus(_) => TelnetOption::MSSP,
265            TelnetArgument::TerminalType(_) => TelnetOption::TTYPE,
266            TelnetArgument::Unknown(option, _) => TelnetOption::Unknown(option.to_u8()),
267        }
268    }
269}
270
271impl std::fmt::Display for TelnetArgument {
272    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
273        match self {
274            TelnetArgument::NAWSWindowSize(v) => write!(f, "{v}"),
275            TelnetArgument::CharsetRequest(v) => write!(f, "CharsetRequest({v:?})"),
276            TelnetArgument::CharsetAccepted(v) => write!(f, "CharsetAccepted({v:?})"),
277            TelnetArgument::CharsetRejected => write!(f, "CharsetRejected"),
278            TelnetArgument::CharsetTTableRejected => write!(f, "CharsetTableRejected"),
279            TelnetArgument::GMCP(msg) => write!(f, "GMCP({msg})"),
280            TelnetArgument::MudServerData(msg) => write!(f, "MudServerData({msg})"),
281            TelnetArgument::MudServerStatus(msg) => write!(f, "MudServerStatus({msg})"),
282            TelnetArgument::TerminalType(ttype) => write!(f, "TerminalType({ttype})"),
283            TelnetArgument::Unknown(o, v) => write!(f, "{o}-{v:?}"),
284        }
285    }
