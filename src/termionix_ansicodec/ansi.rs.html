25/// Ansi Sequence represents a series of bytes read from a [TelnetCodec] which translates to a valid
26/// Ansi Sequence. Sequences include individual characters, control commands, etc.
27#[derive(Clone, Debug, PartialEq, Eq)]
28pub enum AnsiSequence {
29    /// A single ASCII character in the range 0x20-0x7E (printable ASCII).
30    ///
31    /// This excludes escape sequences (ESC), control codes, and multi-byte UTF-8
32    /// characters. These are standard printable ASCII characters that can be
33    /// directly rendered or processed as text.
34    Character(char),
35
36    /// A multi-byte UTF-8 encoded Unicode character.
37    ///
38    /// This is returned after successfully parsing a 2-4 byte UTF-8 sequence.
39    /// Characters in the range U+0080 and above are represented here. Invalid
40    /// UTF-8 sequences are replaced with the Unicode replacement character U+FFFD.
41    ///
42    /// # UTF-8 Byte Sequences
43    /// - 2-byte: 0xC0-0xDF (followed by 1 continuation byte)
44    /// - 3-byte: 0xE0-0xEF (followed by 2 continuation bytes)
45    /// - 4-byte: 0xF0-0xF7 (followed by 3 continuation bytes)
46    Unicode(char),
47
48    /// A C0 or C1 control character.
49    ///
50    /// These are non-printable control codes in the ranges:
51    /// - C0: 0x00-0x1F (excluding ESC at 0x1B)
52    /// - DEL: 0x7F
53    /// - C1: 0x80-0x9F
54    ///
55    /// Common examples include NULL (0x00), Bell (0x07), Backspace (0x08),
56    /// Tab (0x09), Line Feed (0x0A), and Carriage Return (0x0D).
57    AnsiControlCode(AnsiControlCode),
58
59    /// A standalone ESC character (0x1B) that is not part of a recognized sequence.
60    ///
61    /// This occurs when an ESC character is followed by a byte that doesn't
62    /// initiate a known ANSI escape sequence. The ESC was not consumed as part
63    /// of a control sequence.
64    AnsiEscape,
65
66    /// Control Sequence Introducer - a general CSI escape sequence.
67    ///
68    /// Format: `ESC [ <params> <final_byte>`
69    ///
70    /// CSI sequences are used for cursor movement, screen manipulation, and other
71    /// terminal control operations. The final byte (0x40-0x7E) determines the
72    /// specific command. Common examples:
73    /// - `ESC[H` - Cursor Home
74    /// - `ESC[2J` - Clear Screen
75    /// - `ESC[10;20H` - Move cursor to row 10, column 20
76    ///
77    /// Note: SGR sequences (ending with 'm') are parsed separately and returned
78    /// as the `SGR` variant instead.
79    AnsiCSI(AnsiControlSequenceIntroducer),
80
81    /// Select Graphic Rendition - a specialized CSI sequence for text styling.
82    ///
83    /// Format: `ESC [ <params> m`
84    ///
85    /// SGR sequences control text appearance including colors, bold, italic,
86    /// underline, and other visual attributes. This is a specialized form of
87    /// CSI sequence that is parsed into a `Style` object for convenience.
88    ///
89    /// Examples:
90    /// - `ESC[0m` - Reset all attributes
91    /// - `ESC[1m` - Bold
92    /// - `ESC[31m` - Red foreground
93    /// - `ESC[1;31;42m` - Bold red text on a green background
94    AnsiSGR(AnsiSelectGraphicRendition),
95
96    /// Operating System Command - a sequence for terminal-specific operations.
97    ///
98    /// Format: `ESC ] <params> ST` or `ESC ] <params> BEL`
99    ///
100    /// OSC sequences communicate with the terminal emulator to perform operations
101    /// like setting the window title, changing color palettes, or other OS-level
102    /// terminal features. The sequence is terminated by either ST (String Terminator,
103    /// ESC \) or BEL (0x07).
104    ///
105    /// The raw bytes (excluding the terminator) are returned for interpretation
106    /// by the application.
107    AnsiOSC(AnsiOperatingSystemCommand),
108
109    /// Device Control String - a sequence for device-specific control.
110    ///
111    /// Format: `ESC P <params> ST`
112    ///
113    /// DCS sequences are used to send device-specific control strings to the
114    /// terminal. They are terminated by ST (ESC \). The contents are device-
115    /// dependent and returned as raw bytes.
116    AnsiDCS(AnsiDeviceControlString),
117
118    /// Start of String - a legacy control sequence.
119    ///
120    /// Format: `ESC X <data> ST`
121    ///
122    /// SOS is a rarely used control function from ISO 6429. It marks the start
123    /// of a control string that is terminated by ST (ESC \). The contents are
124    /// returned as raw bytes.
125    AnsiSOS(AnsiStartOfString),
126
127    /// String Terminator - marks the end of a string control sequence.
128    ///
129    /// Format: `ESC \`
130    ///
131    /// ST is used to terminate string-type control sequences (OSC, DCS, SOS, PM, APC).
132    /// When encountered outside of a string sequence context, it's returned as a
133    /// standalone result with empty data.
134    AnsiST,
135
136    /// Privacy Message - a control sequence for private data.
137    ///
138    /// Format: `ESC ^ <data> ST`
139    ///
140    /// PM is a control function from ISO 6429 used to delimit privacy messages.
141    /// The sequence is terminated by ST (ESC \) and the contents are returned
142    /// as raw bytes.
143    AnsiPM(AnsiPrivacyMessage),
144
145    /// Application Program Command - a control sequence for application-specific commands.
146    ///
147    /// Format: `ESC _ <data> ST`
148    ///
149    /// APC sequences allow applications to send custom commands through the
150    /// terminal. The sequence is terminated by ST (ESC \), and the contents are
151    /// returned as raw bytes for application-specific interpretation.
152    AnsiAPC(AnsiApplicationProgramCommand),
153
154    /// Passthrough Telnet Command
155    TelnetCommand(TelnetCommand),
156}

158impl AnsiSequence {
159    /// Returns the encoded byte length of this ANSI sequence.
160    ///
161    /// This method calculates the total number of bytes that will be produced when this
162    /// sequence is encoded to its byte representation. The length includes all escape codes,
163    /// parameters, and terminators, but represents the wire format size, not the semantic content.
164    ///
165    /// # Returns
166    ///
167    /// The total byte length in the encoded form. For simple sequences like a single character,
168    /// this matches the character's UTF-8 byte length. For complex escape sequences, this includes
169    /// all prefix, parameter, and suffix bytes.
170    ///
171    /// # Performance
172    ///
173    /// This is an O(1) or O(n) operation depending on variant:
174    /// - **O(1)** for variants with fixed sizes (Character, Unicode, AnsiEscape, AnsiST)
175    /// - **O(n)** for variants with dynamic sizes that delegate to nested types (Control, CSI, SGR, etc.)
176    ///   where n is the complexity of the nested structure
177    ///
178    /// # Examples
179    ///
180    /// ```rust
181    /// // TODO: Fix this Test
182    /// use termionix_ansicodec::{AnsiSequence, AnsiControlCode, AnsiSelectGraphicRendition};
183    ///
184    /// // Single ASCII character
185    /// let seq = AnsiSequence::Character('A');
186    /// assert_eq!(seq.len(), 1);
187    ///
188    /// // Unicode character (3 bytes)
189    /// let seq = AnsiSequence::Unicode('ä¸–');
190    /// assert_eq!(seq.len(), 3);
191    ///
192    /// // Control code (1 byte)
193    /// let seq = AnsiSequence::Control(AnsiControlCode::BEL);
194    /// assert_eq!(seq.len(), 1);
195    ///
196    /// // Escape sequence (5 bytes: ESC [ 3 1 m)
197    /// let seq = AnsiSequence::AnsiSGR(AnsiSelectGraphicRendition::Unknown(vec![b'3', b'1']));
198    /// assert_eq!(seq.len(), 4 + 2); // ESC [ ... m + data
199    /// ```
200    ///
201    /// # Use Cases
202    ///
203    /// - **Buffer allocation**: Pre-allocate buffers for encoding before calling `encode()`
204    /// - **Progress tracking**: Calculate total bytes to be written
205    /// - **Wire sidechannel**: Determine message sizes for network transmission
206    /// - **Statistics**: Analyze the size distribution of sequences
207    pub fn len(&self) -> usize {
208        match self {
209            AnsiSequence::Character(c) => c.len_utf8(),
210            AnsiSequence::Unicode(c) => c.len_utf8(),
211            AnsiSequence::AnsiControlCode(code) => code.len(),
212            AnsiSequence::AnsiEscape => 1,
213            AnsiSequence::AnsiCSI(csi) => csi.len(),
214            AnsiSequence::AnsiSGR(sgr) => sgr.len(None),
215            AnsiSequence::AnsiOSC(osc) => osc.len(),
216            AnsiSequence::AnsiDCS(dcs) => dcs.len(),
217            AnsiSequence::AnsiSOS(sos) => sos.len(),
218            AnsiSequence::AnsiST => 2, // ESC \
219            AnsiSequence::AnsiPM(pm) => pm.len(),
220            AnsiSequence::AnsiAPC(apc) => apc.len(),
221            AnsiSequence::TelnetCommand(cmd) => cmd.len(),
222        }
223    }

225    /// Encode this ANSI sequence to a `BufMut` buffer.
226    ///
227    /// This method encodes the sequence into its byte representation and writes it to the
228    /// provided mutable buffer. The buffer will be advanced by the number of bytes written.
229    /// This is the preferred method for encoding into buffers that implement `BufMut`.
230    ///
231    /// # Arguments
232    ///
233    /// * `dst` - A mutable buffer implementing `BufMut` (e.g., `BytesMut`, `Vec<u8>` via
234    ///   `BufMut::writer()`, etc.). The buffer is advanced by the bytes written.
235    ///
236    /// # Returns
237    ///
238    /// Returns `Ok(bytes_written)` on success, containing the exact number of bytes that were
239    /// written to the buffer. Returns an `AnsiResult` error if encoding fails (rare, typically
240    /// only due to I/O errors in the underlying writer).
241    ///
242    /// # Performance
243    ///
244    /// This method is O(n) where n is the length of the encoded sequence. It performs a single
245    /// pass through the data, writing directly to the buffer without intermediate allocations.
246    ///
247    /// # Buffer Behavior
248    ///
249    /// - The buffer is only written to; existing content is preserved
250    /// - The buffer's write position is advanced automatically
251    /// - No bounds checking is performed; the buffer must have sufficient capacity
252    /// - For `BytesMut`, automatic growth occurs if needed
253    ///
254    /// # Examples
255    ///
256    /// ## Basic Encoding
257    ///
258    /// ```rust
259    /// use termionix_ansicodec::ansi::AnsiSequence;
260    /// use bytes::BytesMut;
261    ///
262    /// let seq = AnsiSequence::Character('X');
263    /// let mut buffer = BytesMut::new();
264    /// let written = seq.encode(&mut buffer).unwrap();
265    /// assert_eq!(written, 1);
266    /// assert_eq!(&buffer[..], b"X");
267    /// ```
268    ///
269    /// ## Encoding Multiple Sequences
270    ///
271    /// ```rust
272    /// use termionix_ansicodec::{AnsiSequence, ansi::AnsiControlCode};
273    /// use bytes::BytesMut;
274    ///
275    /// let mut buffer = BytesMut::new();
276    ///
277    /// // Encode "Hello"
278    /// for c in "Hello".chars() {
279    ///     AnsiSequence::Character(c).encode(&mut buffer).unwrap();
280    /// }
281    ///
282    /// // Encode a newline
283    /// AnsiSequence::Control(AnsiControlCode::LF).encode(&mut buffer).unwrap();
284    ///
285    /// assert_eq!(&buffer[..], b"Hello\n");
286    /// ```
287    ///
288    /// ## Encoding ANSI Escape Sequences
289    ///
290    /// ```rust
291    /// // TODO: Fix this example
292    /// use termionix_ansicodec::{AnsiSequence, ansi::AnsiSelectGraphicRendition};
293    /// use bytes::BytesMut;
294    ///
295    /// let sgr = AnsiSelectGraphicRendition::parse(vec![31]); // Red
296    /// let seq = AnsiSequence::AnsiSGR(sgr);
297    /// let mut buffer = BytesMut::new();
298    /// seq.encode(&mut buffer).unwrap();
299    /// // Result: ESC[31m
300    /// ```
301    ///
302    /// # Error Handling
303    ///
304    /// While encoding to `BufMut` rarely fails, errors can occur if:
305    /// - The underlying buffer is unable to allocate additional capacity
306    /// - The writer encounters an I/O error (rare for in-memory buffers)
307    ///
308    /// # See Also
309    ///
310    /// - [`write()`](AnsiSequence::write) - Write to a `std::io::Write` trait object
311    /// - [`len()`](AnsiSequence::len) - Get the encoded byte length without encoding
312    /// - [`Display`](std::fmt::Display) - Convert to a string representation
313    pub fn encode<T: BufMut>(&self, dst: &mut T) -> AnsiCodecResult<usize> {
314        Ok(self.write(&mut dst.writer())?)
315    }

317    /// Write this ANSI sequence to a `std::io::Write` writer.
318    ///
319    /// This method performs the actual encoding of the sequence into its byte representation
320    /// and writes the bytes to the provided writer. This is the low-level method that both
321    /// `encode()` and `Display` delegate to internally.
322    ///
323    /// # Arguments
324    ///
325    /// * `writer` - A mutable writer implementing `std::io::Write` (e.g., `File`, `BufWriter`,
326    ///   `Vec<u8>`, a socket, etc.). The writer's internal position is advanced by the bytes
327    ///   written.
328    ///
329    /// # Returns
330    ///
331    /// Returns `Ok(bytes_written)` on success, containing the exact number of bytes written.
332    /// Returns `std::io::Error` if the writer fails (e.g., disk full, broken pipe, etc.).
333    ///
334    /// # Performance
335    ///
336    /// This is O(n) where n is the length of the encoded sequence. It performs one or more
337    /// `write_all()` calls to the writer, depending on the sequence complexity.
338    ///
339    /// # Behavior by Variant
340    ///
341    /// - **Character**: Encodes as 1-4 bytes (UTF-8 encoded character)
342    /// - **Unicode**: Encodes as 2-4 bytes (UTF-8 encoded character)
343    /// - **Control**: Encodes as 1 byte (the control code byte)
344    /// - **AnsiEscape**: Writes the ESC byte (0x1B)
345    /// - **AnsiCSI**: Delegates to the CSI command's `write()` method
346    /// - **AnsiSGR**: Writes `ESC [ <data> m` format
347    /// - **AnsiOSC**: Writes `ESC ] <data> ST` format
348    /// - **AnsiDCS**: Writes `ESC P <data> ST` format
349    /// - **AnsiSOS**: Writes `ESC X <data> ST` format
350    /// - **AnsiST**: Writes the ST terminator sequence
351    /// - **AnsiPM**: Writes `ESC ^ <data> ST` format
352    /// - **AnsiAPC**: Writes `ESC _ <data> ST` format
353    /// - **TelnetCommand**: Delegates to the Telnet command's `write()` method
354    ///
355    /// # Examples
356    ///
357    /// ## Writing to a Vector
358    ///
359    /// ```rust
360    /// use termionix_ansicodec::AnsiSequence;
361    ///
362    /// let seq = AnsiSequence::Character('A');
363    /// let mut output = Vec::new();
364    /// let written = seq.write(&mut output).unwrap();
365    /// assert_eq!(written, 1);
366    /// assert_eq!(output, b"A");
367    /// ```
368    ///
369    /// ## Writing to a File
370    ///
371    /// ```rust
372    /// use termionix_ansicodec::AnsiSequence;
373    /// use std::io::Cursor;
374    ///
375    /// let seq = AnsiSequence::Character('X');
376    /// let mut cursor = Cursor::new(Vec::new());
377    /// let written = seq.write(&mut cursor).unwrap();
378    /// assert_eq!(written, 1);
379    /// ```
380    ///
381    /// ## Writing Multiple Sequences
382    ///
383    /// ```rust
384    /// use termionix_ansicodec::{AnsiSequence, AnsiControlCode};
385    ///
386    /// let mut output = Vec::new();
387    ///
388    /// AnsiSequence::Character('H').write(&mut output).unwrap();
389    /// AnsiSequence::Character('i').write(&mut output).unwrap();
390    /// AnsiSequence::Control(AnsiControlCode::LF).write(&mut output).unwrap();
391    ///
392    /// assert_eq!(output, b"Hi\n");
393    /// ```
394    ///
395    /// ## Streaming to stdout
396    ///
397    /// ```rust
398    /// use termionix_ansicodec::AnsiSequence;
399    /// use std::io::stdout;
400    ///
401    /// let seq = AnsiSequence::Character('!');
402    /// // This would write to the terminal
403    /// // seq.write(&mut stdout()).unwrap();
404    /// ```
405    ///
406    /// # Error Handling
407    ///
408    /// Errors from the writer are propagated directly. Common errors include:
409    /// - `io::ErrorKind::WriteZero`: Writer refused to write any bytes
410    /// - `io::ErrorKind::Interrupted`: I/O operation was interrupted
411    /// - `io::ErrorKind::BrokenPipe`: Broken pipe when writing to a process or network
412    /// - `io::ErrorKind::PermissionDenied`: No write permission on the target
413    ///
414    /// # Performance Considerations
415    ///
416    /// - For repeated writes to the same destination, consider wrapping in `BufWriter`
417    /// - Writing to `Vec<u8>` is typically faster than writing to disk or network
418    /// - The method makes minimal allocations (typically none for fixed-size sequences)
419    ///
420    /// # See Also
421    ///
422    /// - [`encode()`](AnsiSequence::encode) - Encode to a `BufMut` buffer
423    /// - [`len()`](AnsiSequence::len) - Get the byte length without writing
424    /// - [`Display`](std::fmt::Display) - Convert to a string representation
425    pub fn write<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<usize> {
426        match self {
427            AnsiSequence::Character(c) => {
428                let mut buf = [0u8; 4];
429                let encoded = c.encode_utf8(&mut buf);
430                writer.write_all(encoded.as_bytes())?;
431                Ok(encoded.len())
432            }
433            AnsiSequence::Unicode(c) => {
434                let mut buf = [0u8; 4];
435                let encoded = c.encode_utf8(&mut buf);
436                writer.write_all(encoded.as_bytes())?;
437                Ok(encoded.len())
438            }
439            AnsiSequence::AnsiControlCode(code) => code.write(writer),
440            AnsiSequence::AnsiEscape => {
441                writer.write_all(&[0x1B])?;
442                Ok(1)
443            }
444            AnsiSequence::AnsiCSI(csi) => csi.write(writer),
445            AnsiSequence::AnsiSGR(sgr) => sgr.write(writer, None),
446            AnsiSequence::AnsiOSC(osc) => osc.write(writer),
447            AnsiSequence::AnsiDCS(dcs) => dcs.write(writer),
448            AnsiSequence::AnsiSOS(sos) => sos.write(writer),
449            AnsiSequence::AnsiST => {
450                writer.write_all(&[0x1B, 0x5C])?;
451                Ok(2)
452            }
453            AnsiSequence::AnsiPM(pm) => pm.write(writer),
454            AnsiSequence::AnsiAPC(apc) => apc.write(writer),
455            AnsiSequence::TelnetCommand(cmd) => cmd.write(writer),
456        }
457    }
458}

480/// Telnet command types
481///
482/// This enum represents various Telnet sidechannel commands as defined in RFC 854 and related RFCs.
483/// Telnet commands are used to control the connection and request specific operations from the
484/// remote host or terminal.
485///
486/// # Format
487///
488/// Telnet commands are typically encoded as `IAC <command>` where IAC is the Interpret As Command
489/// byte (0xFF). Some commands (like Negotiation and Subnegotiation) require additional parameters.
490#[derive(Clone, Debug, Eq, PartialEq)]
491pub enum TelnetCommand {
492    /// No Operation - Does nothing but sends an IAC command.
493    ///
494    /// Format: `IAC NOP` (0xFF 0xF1)
495    ///
496    /// This command is often used as a keep-alive or to test the connection.
497    NoOperation,
498
499    /// End of urgent Data Stream - Marks the end of urgent data.
500    ///
501    /// Format: `IAC DM` (0xFF 0xF2)
502    ///
503    /// Used to synchronize urgent and normal data streams in the connection.
504    DataMark,
505
506    /// Break - Operator pressed the Break key or the Attention key.
507    ///
508    /// Format: `IAC BRK` (0xFF 0xF3)
509    ///
510    /// Sends an interrupt signal to the remote system, typically used to interrupt
511    /// a running process on the server.
512    Break,
513
514    /// Interrupt Process - Request immediate interrupt of the current process.
515    ///
516    /// Format: `IAC IP` (0xFF 0xF4)
517    ///
518    /// Sends a stronger interrupt signal than Break, typically mapped to SIGINT in Unix-like systems.
519    InterruptProcess,
520
521    /// Cancel Output - Request that the remote system cancel output to the client.
522    ///
523    /// Format: `IAC AO` (0xFF 0xF5)
524    ///
525    /// Used when the client no longer wants to receive output from the remote process.
526    AbortOutput,
527
528    /// Are You There - Request acknowledgment from the remote system.
529    ///
530    /// Format: `IAC AYT` (0xFF 0xF6)
531    ///
532    /// Used to test if the connection is still alive or if the remote host is responding.
533    AreYouThere,
534
535    /// Erase Character - Request that the operator erase the previous character.
536    ///
537    /// Format: `IAC EC` (0xFF 0xF7)
538    ///
539    /// Equivalent to sending a backspace or delete request to the remote system.
540    EraseCharacter,
541
542    /// Erase Line - Request that the operator erase the previous line.
543    ///
544    /// Format: `IAC EL` (0xFF 0xF8)
545    ///
546    /// Requests the remote system to clear the current input line.
547    EraseLine,
548
549    /// Go Ahead - End of input for half-duplex connections.
550    ///
551    /// Format: `IAC GA` (0xFF 0xF9)
552    ///
553    /// Used in half-duplex mode to indicate that the sender has finished transmitting
554    /// and the receiver may now send data. Rarely used in modern systems.
555    GoAhead,
556
557    /// End of Record - Marks the end of a prompt.
558    ///
559    /// Format: `IAC EOR` (0xFF 0xEF)
560    ///
561    /// Used by MUD servers to mark the end of a prompt. A prompt is considered
562    /// any line that does not end with \r\n. This allows clients to distinguish
563    /// between regular output and prompts that require user input.
564    EndOfRecord,
565
566    /// Telnet Negotiation Result.
567    ///
568    /// # Arguments
569    ///
570    /// - `TelnetOption` - The option being negotiated (e.g., ECHO, LINEMODE, etc.)
571    /// - `OptionSide` - Which side the option is being updated
572    /// - `bool` - Enabled or Not
573    OptionStatus(TelnetOption, TelnetSide, bool),
574
575    /// Telnet Subnegotiation - Send subnegotiation data for a negotiated option.
576    ///
577    /// Format: `IAC SB <option> <data...> IAC SE` (0xFF 0xFA <option> <data...> 0xFF 0xF0)
578    ///
579    /// # Arguments
580    ///
581    /// - `TelnetArgument` - Contains the option and associated data for that option
582    ///
583    /// After an option is negotiated via Negotiation, subnegotiation allows sending
584    /// additional parameters or data specific to that option.
585    Subnegotiation(TelnetArgument),
586}

588impl TelnetCommand {
589    /// Returns the encoded byte length of this Telnet command.
590    ///
591    /// This method calculates the number of bytes that will be produced when the command
592    /// is encoded to its byte representation. Most Telnet commands are encoded as `IAC <command>`
593    /// (3 bytes total: 0xFF + command_byte), while Subnegotiation commands have variable
594    /// length depending on their data payload.
595    ///
596    /// # Returns
597    ///
598    /// The number of bytes in the encoded form:
599    /// - Simple commands: 3 bytes (IAC + command + option/parameter)
600    /// - Subnegotiation: 2 + data_length + 2 (IAC SB + option + data + IAC SE)
601    ///
602    /// # Performance
603    ///
604    /// This is O(1) for simple commands and O(n) for Subnegotiation where n is the data length.
605    ///
606    /// # Examples
607    ///
608    /// ```rust
609    /// use termionix_ansicodec::ansi::TelnetCommand;
610    ///
611    /// // Simple commands are always 3 bytes
612    /// assert_eq!(TelnetCommand::NoOperation.len(), 3);
613    /// assert_eq!(TelnetCommand::Break.len(), 3);
614    /// assert_eq!(TelnetCommand::AreYouThere.len(), 3);
615    ///
616    /// // Subnegotiation includes the data length
617    /// // Subnegotiation(arg) => 2 (IAC SB) + 1 (option) + arg.data().len() + 2 (IAC SE)
618    /// ```
619    ///
620    /// # Use Cases
621    ///
622    /// - **Buffer allocation**: Pre-allocate space before encoding
623    /// - **Protocol implementation**: Calculate message sizes for Telnet sidechannel
624    /// - **Stream management**: Track total bytes in a command sequence
625    pub fn len(&self) -> usize {
626        match self {
627            TelnetCommand::NoOperation => 3,
628            TelnetCommand::DataMark => 3,
629            TelnetCommand::Break => 3,
630            TelnetCommand::InterruptProcess => 3,
631            TelnetCommand::AbortOutput => 3,
632            TelnetCommand::AreYouThere => 3,
633            TelnetCommand::EraseCharacter => 3,
634            TelnetCommand::EraseLine => 3,
635            TelnetCommand::GoAhead => 3,
636            TelnetCommand::EndOfRecord => 3,
637            TelnetCommand::OptionStatus(_, _, _) => 0,
638            TelnetCommand::Subnegotiation(arg) => 2 + arg.len(),
639        }
640    }

656    /// # Wire Format
657    ///

659    /// - **Simple commands**: `0xFF <command_byte>` (3 bytes total with optional parameter)
660    /// - **Subnegotiation**: `0xFF 0xFA <option> <data...> 0xFF 0xF0`

682    /// Write this Telnet command to a `std::io::Write` writer.
683    ///
684    /// This method performs the actual encoding of the Telnet command into its wire format
685    /// and writes it to the provided writer. This is the low-level method that `encode()` delegates to.
686    ///
687    /// # Arguments
688    ///
689    /// * `writer` - A mutable writer implementing `std::io::Write`
690    ///
691    /// # Returns
692    ///
693    /// Returns `Ok(bytes_written)` on success, containing the number of bytes written.
694    /// Returns `std::io::Error` if writing fails.
695    ///
696    /// # Encoding Behavior
697    ///
698    /// Each command is encoded according to RFC 854:
699    ///
700    /// - **NoOperation**: `IAC NOP` â†’ `0xFF 0xF1`
701    /// - **DataMark**: `IAC DM` â†’ `0xFF 0xF2`
702    /// - **Break**: `IAC BRK` â†’ `0xFF 0xF3`
703    /// - **InterruptProcess**: `IAC IP` â†’ `0xFF 0xF4`
704    /// - **AbortOutput**: `IAC AO` â†’ `0xFF 0xF5`
705    /// - **AreYouThere**: `IAC AYT` â†’ `0xFF 0xF6`
706    /// - **EraseCharacter**: `IAC EC` â†’ `0xFF 0xF7`
707    /// - **EraseLine**: `IAC EL` â†’ `0xFF 0xF8`
708    /// - **GoAhead**: `IAC GA` â†’ `0xFF 0xF9`
709    /// - **Negotiation**: `IAC <request> <option>` â†’ `0xFF <request_byte> <option_byte>`
710    /// - **Subnegotiation**: `IAC SB <option> <data...> IAC SE` â†’ `0xFF 0xFA <option> <data...> 0xFF 0xF0`
711    ///
712    /// # Examples
713    ///
714    /// ```rust
715    /// use termionix_ansicodec::ansi::TelnetCommand;
716    ///
717    /// let cmd = TelnetCommand::AreYouThere;
718    /// let mut output = Vec::new();
719    /// let written = cmd.write(&mut output).unwrap();
720    /// assert_eq!(written, 3);
721    /// assert_eq!(output, vec![0xFF, 0xF6]);
722    /// ```
723    ///
724    /// # See Also
725    ///
726    /// - [`encode()`](TelnetCommand::encode) - Encode to a `BufMut` buffer
727    /// - [`len()`](TelnetCommand::len) - Get the encoded byte length
728    pub fn write<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<usize> {
729        match self {
730            TelnetCommand::NoOperation => {
731                writer.write_all(&[0xFF, 0xF1])?;
732                Ok(3)
733            }
734            TelnetCommand::DataMark => {
735                writer.write_all(&[0xFF, 0xF2])?;
736                Ok(3)
737            }
738            TelnetCommand::Break => {
739                writer.write_all(&[0xFF, 0xF3])?;
740                Ok(3)
741            }
742            TelnetCommand::InterruptProcess => {
743                writer.write_all(&[0xFF, 0xF4])?;
744                Ok(3)
745            }
746            TelnetCommand::AbortOutput => {
747                writer.write_all(&[0xFF, 0xF5])?;
748                Ok(3)
749            }
750            TelnetCommand::AreYouThere => {
751                writer.write_all(&[0xFF, 0xF6])?;
752                Ok(3)
753            }
754            TelnetCommand::EraseCharacter => {
755                writer.write_all(&[0xFF, 0xF7])?;
756                Ok(3)
757            }
758            TelnetCommand::EraseLine => {
759                writer.write_all(&[0xFF, 0xF8])?;
760                Ok(3)
761            }
762            TelnetCommand::GoAhead => {
763                writer.write_all(&[0xFF, 0xF9])?;
764                Ok(3)
765            }
766            TelnetCommand::EndOfRecord => {
767                writer.write_all(&[0xFF, 0xEF])?;
768                Ok(3)
769            }
770            TelnetCommand::OptionStatus(_option, _side, _enabled) => Ok(3),
771            TelnetCommand::Subnegotiation(arg) => {
772                writer.write_all(&[0xFF, 0xFA])?;
773                writer.write_all(&[arg.option().to_u8()])?;
774                let len = arg.write(writer)?;
775                writer.write_all(&[0xFF, 0xF0])?;
776                Ok(5 + len)
777            }
778        }
779    }

2446    /// - ST: 0x1B 0x5C (String Terminator)
2447    ///
2448    /// # Examples
2449    ///
2450    /// ```rust
2451    /// use bytes::BytesMut;
2452    /// use termionix_ansicodec::ansi::AnsiApplicationProgramCommand;
2453    ///
2454    /// let apc = AnsiApplicationProgramCommand::Unknown(b"command".to_vec());
2455    /// let mut buffer = BytesMut::new();
2456    /// apc.encode(&mut buffer).unwrap();
2457    /// ```
2458    ///
2459    /// # See Also
2460    ///
2461    /// - [`write()`](AnsiApplicationProgramCommand::write) - Write to a `std::io::Write` trait object
2462    pub fn encode<T: BufMut>(&self, dst: &mut T) -> AnsiCodecResult<usize> {
2463        Ok(self.write(&mut dst.writer())?)
2464    }
2465
2466    /// Write this APC sequence to a `std::io::Write` writer.
2467    ///
2468    /// Performs the actual encoding and writes to the provided writer.
2469    ///
2470    /// # Arguments
2471    ///
2472    /// * `writer` - A mutable writer implementing `std::io::Write`
2473    ///
2474    /// # Returns
2475    ///
2476    /// Returns `Ok(bytes_written)` on success.
2477    pub fn write<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<usize> {
2478        match self {
2479            AnsiApplicationProgramCommand::Unknown(data) => {
2480                writer.write_all(b"\x1b_")?;
2481                writer.write_all(data)?;
2482                writer.write_all(b"\x1b\\")?;
2483                Ok(4 + data.len())
2484            }
2485        }
2486    }
2487}

2490    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
2491        match self {
2492            AnsiApplicationProgramCommand::Unknown(data) => {
2493                write!(f, "\x1b_")?;
2494                if let Ok(s) = std::str::from_utf8(data) {
2495                    write!(f, "{}", s)?;
2496                }
2497                write!(f, "\x1b\\")
2498            }
2499        }
2500    }
