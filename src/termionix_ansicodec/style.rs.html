20/// Represents a text style with various formatting attributes and colors for terminal output.
21///
22/// `Style` encapsulates all ANSI terminal styling properties, including text attributes
23/// (bold, italic, underline, etc.) and foreground/background colors. It provides methods
24/// to write ANSI escape sequences to apply these styles in terminal environments.
25///
26/// # Examples
27///
28/// ## Basic Usage
29///
30/// ```rust
31/// use termionix_ansicodec::{Style, Color, ColorMode, Intensity, AnsiConfig};
32/// use std::fmt::Write;
33///
34/// let config = AnsiConfig::enabled();
35/// let mut style = Style::default();
36/// style.intensity = Some(Intensity::Bold);
37/// style.foreground = Some(Color::Red);
38///
39/// let mut output = String::new();
40/// style.write_style(&mut output, Some(&config)).unwrap();
41/// write!(&mut output, "Bold Red Text").unwrap();
42/// Style::write_reset(&mut output, Some(&config)).unwrap();
43/// ```
44///
45/// ## Complex Styling
46///
47/// ```rust
48/// use termionix_ansicodec::{Underline,Style, Color, ColorMode, Intensity};
49///
50/// let style = Style {
51///     intensity: Some(Intensity::Bold),
52///     italic: Some(true),
53///     underline: Some(Underline::Single),
54///     foreground: Some(Color::RGB(255, 100, 50)),
55///     background: Some(Color::Blue),
56///     ..Default::default()
57/// };
58/// ```
59///
60/// # ANSI Escape Codes
61///
62/// The `Style` struct generates ANSI escape sequences in the format `\x1b[<codes>m`,
63/// where `<codes>` is a semicolon-separated list of numeric codes:
64///
65/// - Text attributes: `1` (bold), `2` (dim), `3` (italic), `4` (underline),
66///   `5` (blink), `7` (reverse), `8` (hidden), `9` (strikethrough)
67/// - Foreground colors: `30-37` (basic), `38;5;<n>` (256-color), `38;2;<r>;<g>;<b>` (RGB)
68/// - Background colors: `40-47` (basic), `48;5;<n>` (256-color), `48;2;<r>;<g>;<b>` (RGB)
69/// - Reset: `0` (clear all styles)
70///
71/// # Color Mode Support
72///
73/// Style rendering respects the `ColorMode` setting:
74/// - `ColorMode::None`: No ANSI codes are written
75/// - `ColorMode::Basic`: Basic 8-color support (codes 30-37, 40-47)
76/// - `ColorMode::FixedColor`: 256-color support
77/// - `ColorMode::TrueColor`: Full 24-bit RGB color support
78///
79/// # Default Values
80///
81/// The default style has no formatting attributes enabled and no colors set:
82///
83/// ```rust
84/// use termionix_ansicodec::{Color, Style};
85///
86/// let style = Style::default();
87/// assert_eq!(style.intensity, None);
88/// assert_eq!(style.italic, None);
89/// assert_eq!(style.foreground, None);
90/// assert_eq!(style.background, None);
91/// ```

240    pub fn len(&self, _color_mode: Option<ColorMode>) -> usize {
241        let mut length = 0;
242        let mut code_count = 0;
243
244        // Count Intensity (Bold `1` or Dim/Faint `2` or Normal `22`)
245        match self.intensity {
246            Some(intensity) => match intensity {
247                Intensity::Bold | Intensity::Dim => {
248                    code_count += 1;
249                    length += 1;
250                }
251                Intensity::Normal => {}
252            },
253            None => {}
254        }
255
256        // Count Italic (Enabled `3` or Disabled `23`)
257        match self.italic {
258            Some(italic) => match italic {
259                true => {
260                    code_count += 1;
261                    length += 1;
262                }
263                false => {
264                    code_count += 1;
265                    length += 2;
266                }
267            },
268            None => {}
269        }
270
271        // Count Underline (Single `4`, Double `21`, or Disabled `24`)
272        match self.underline {
273            Some(underline) => match underline {
274                Underline::Single => {
275                    code_count += 1;
276                    length += 1;
277                }
278                Underline::Double => {
279                    code_count += 1;
280                    length += 2;
281                }
282                Underline::Disabled => {
283                    code_count += 1;
284                    length += 2;
285                }
286            },
287            None => {}
288        }
289
290        // Count Blink (slow, rapid, or reset to none)
291        match self.blink {
292            Some(blink) => match blink {
293                Blink::Slow => {
294                    code_count += 1;
295                    length += 1;
296                }
297                Blink::Rapid => {
298                    code_count += 1;
299                    length += 1;
300                }
301                Blink::Off => {
302                    code_count += 1;
303                    length += 2;
304                }
305            },
306            None => {}
307        }
308
309        // Count Reverse (Enabled `7` or Disabled `27`)
310        match self.reverse {
311            Some(reverse) => match reverse {
312                true => {
313                    code_count += 1;
314                    length += 1;
315                }
316                false => {
317                    code_count += 1;
318                    length += 2;
319                }
320            },
321            None => {}
322        }
323
324        // Count Hidden (Enabled `8` or Disabled `28`)
325        match self.hidden {
326            Some(hidden) => match hidden {
327                true => {
328                    code_count += 1;
329                    length += 1;
330                }
331                false => {
332                    code_count += 1;
333                    length += 2;
334                }
335            },
336            None => {}
337        }
338
339        // Count Strike (Enabled `9` or Disabled `29`)
340        match self.hidden {
341            Some(hidden) => match hidden {
342                true => {
343                    code_count += 1;
344                    length += 1;
345                }
346                false => {
347                    code_count += 1;
348                    length += 2;
349                }
350            },
351            None => {}
352        }
353
354        // Count font
355        if self.font.is_some() {
356            code_count += 1;
357            length += 2; // "10"-"20" (all are 2 digits)
358        }
359
360        // Calculate foreground color codes
361        if let Some(fg) = &self.foreground {
362            match fg {
363                Color::Black
364                | Color::Red
365                | Color::Green
366                | Color::Yellow
367                | Color::Blue
368                | Color::Purple
369                | Color::Cyan
370                | Color::White => {
371                    code_count += 1;
372                    length += 2; // "30"-"37"
373                }
374                Color::BrightBlack
375                | Color::BrightRed
376                | Color::BrightGreen
377                | Color::BrightYellow
378                | Color::BrightBlue
379                | Color::BrightPurple
380                | Color::BrightCyan
381                | Color::BrightWhite => {
382                    code_count += 1;
383                    length += 2; // "90"-"97"
384                }
385                Color::Fixed(n) => {
386                    code_count += 3; // "38", "5", and the number
387                    length += 2 + 1 + n.to_string().len(); // "38" + "5" + number
388                }
389                Color::RGB(r, g, b) => {
390                    code_count += 5; // "38", "2", r, g, b
391                    length +=
392                        2 + 1 + r.to_string().len() + g.to_string().len() + b.to_string().len(); // "38" + "2" + r + g + b
393                }
394            }
395        }
396
397        // Calculate background color codes
398        if let Some(bg) = &self.background {
399            match bg {
400                Color::Black
401                | Color::Red
402                | Color::Green
403                | Color::Yellow
404                | Color::Blue
405                | Color::Purple
406                | Color::Cyan
407                | Color::White => {
408                    code_count += 1;
409                    length += 2; // "40"-"47"
410                }
411                Color::BrightBlack
412                | Color::BrightRed
413                | Color::BrightGreen
414                | Color::BrightYellow
415                | Color::BrightBlue
416                | Color::BrightPurple
417                | Color::BrightCyan
418                | Color::BrightWhite => {
419                    code_count += 1;
420                    length += 3; // "100"-"107"
421                }
422                Color::Fixed(n) => {
423                    code_count += 3; // "48", "5", and the number
424                    length += 2 + 1 + n.to_string().len(); // "48" + "5" + number
425                }
426                Color::RGB(r, g, b) => {
427                    code_count += 5; // "48", "2", r, g, b
428                    length +=
429                        2 + 1 + r.to_string().len() + g.to_string().len() + b.to_string().len(); // "48" + "2" + r + g + b
430                }
431            }
432        }
433
434        // Add unknown SGR bytes
435        for byte in &self.unknown {
436            code_count += 1;
437            length += byte.to_u8().to_string().len();
438        }
439
440        if code_count == 0 {
441            return 0; // No style codes, no escape sequence
442        }
443
444        // Calculate total length:
445        // - "\x1b[" = 2 bytes
446        // - "m" = 1 byte
447        // - Semicolons between codes = (code_count - 1) bytes
448        // - Plus the actual length of all codes calculated above
449
450        let semicolons = if code_count > 1 { code_count - 1 } else { 0 };
451
452        2 + length + semicolons + 1
453    }

498    /// Writes ANSI style escape codes to the given writer based on the style attributes
499    /// and colors defined in the instance. This method is used to apply text formatting
500    /// when ANSI color output is enabled.
501    ///
502    /// # Parameters
503    ///
504    /// - `mode`: A reference to a `ColorMode` instance that determines if ANSI colors
505    ///   are supported. If ANSI is not supported, no styling is applied, and the function
506    ///   returns early.
507    /// - `writer`: A mutable reference to a type that implements the `std::fmt::Write`
508    ///   trait, where the ANSI escape sequences will be written.
509    ///
510    /// # Returns
511    ///
512    /// - `std::fmt::Result`: Returns `Ok(())` if the escape codes are successfully written
513    ///   or if no styling/formatting is needed. Returns an error if writing to the writer fails.
514    ///
515    /// # Behavior
516    ///
517    /// - When ANSI coloring is disabled (e.g., `ColorMode::is_ansi` is `false`), the method
518    ///   returns immediately without writing any styles.
519    /// - For each enabled text attribute (e.g., bold, italic, underline, etc.), the corresponding
520    ///   escape code is added to a sequence of codes.
521    /// - If foreground or background colors are specified, they are included in the ANSI escape
522    ///   sequence. Supports standard, fixed 256-color, and RGB color codes.
523    /// - If no attributes or colors are specified, no output is written, and the function exits
524    ///   normally.
525    ///
526    /// # Example
527    ///
528    /// ```rust
529    /// use termionix_ansicodec::{Style, Color, ColorMode, Intensity, Underline, Blink, AnsiConfig};
530    /// use std::fmt::Write;
531    ///
532    /// let config = AnsiConfig::enabled();
533    /// let style = Style {
534    ///     intensity: Some(Intensity::Bold),
535    ///     italic: None,
536    ///     underline: Some(Underline::Single),
537    ///     blink: Some(Blink::Off),
538    ///     reverse: None,
539    ///     hidden: None,
540    ///     strike: None,
541    ///     font: None,
542    ///     script: None,
543    ///     ideogram: None,
544    ///     foreground: Some(Color::Red),
545    ///     background: Some(Color::RGB(10, 20, 30)),
546    ///     unknown: Vec::new(),
547    /// };
548    ///
549    /// let mut output = String::new();
550    /// style.write_style(&mut output, Some(&config)).unwrap();
551    /// assert_eq!(output, "\x1b[1;4;25;31;48;2;10;20;30m");
552    /// ```
553    ///
554    /// # Notes
555    ///
556    /// - This method assumes an implementation of a `Style` struct with text attributes
557    ///   such as bold or underline, as well as `foreground` and `background` fields
558    ///   for colors.
559    /// - The `Color` enum is expected to support standard colors, fixed 256-color values,
560    ///   and custom RGB values.
561    /// ```
562    pub fn write<W: std::io::Write>(
563        &self,
564        writer: &mut W,
565        color_mode: Option<ColorMode>,
566    ) -> std::io::Result<usize> {
567        let codes = self.codes(color_mode);
568
569        if !codes.is_empty() {
570            let sequence = format!("\x1b[{}m", codes.join(";"));
571            writer.write_all(sequence.as_bytes())?;
572            Ok(sequence.len())
573        } else {
574            Ok(0)
575        }
576    }

593    fn codes(&self, color_mode: Option<ColorMode>) -> Vec<String> {
594        // If color mode is None, return empty vector (no ANSI codes)
595        if let Some(ColorMode::None) = color_mode {
596            return Vec::new();
597        }
598
599        let mut codes = Vec::new();
600
601        // Write Intensity (Bold `1` or Dim/Faint `2` or Normal `22`)
602        match self.intensity {
603            Some(intensity) => match intensity {
604                Intensity::Bold => codes.push("1".to_string()),
605                Intensity::Dim => codes.push("2".to_string()),
606                Intensity::Normal => codes.push("22".to_string()),
607            },
608            None => {}
609        }
610
611        // Write Italic (Enabled `3` or Disabled `23`)
612        match self.italic {
613            Some(reverse) => match reverse {
614                true => codes.push("3".to_string()),
615                false => codes.push("23".to_string()),
616            },
617            None => {}
618        }
619

624                Underline::Double => codes.push("21".to_string()),

626            },
627            None => {}
628        }
629
630        // Write Blink (Slow `5`, Rapid `6`, or Off `25`)
631        match self.blink {
632            Some(blink) => match blink {
633                Blink::Slow => codes.push("5".to_string()),
634                Blink::Rapid => codes.push("6".to_string()),
635                Blink::Off => codes.push("25".to_string()),

637            None => {}
638        }

647        }

675                Color::Black => codes.push("30".to_string()),
676                Color::Red => codes.push("31".to_string()),
677                Color::Green => codes.push("32".to_string()),
678                Color::Yellow => codes.push("33".to_string()),
679                Color::Blue => codes.push("34".to_string()),
680                Color::Purple => codes.push("35".to_string()),
681                Color::Cyan => codes.push("36".to_string()),
682                Color::White => codes.push("37".to_string()),
683                Color::BrightBlack => codes.push("90".to_string()),
684                Color::BrightRed => codes.push("91".to_string()),
685                Color::BrightGreen => codes.push("92".to_string()),
686                Color::BrightYellow => codes.push("93".to_string()),
687                Color::BrightBlue => codes.push("94".to_string()),
688                Color::BrightPurple => codes.push("95".to_string()),

748    /// Parses SGR (Select Graphic Rendition) parameters into a Style struct.
749    ///
750    /// This function takes a slice of SGR parameter codes (the numeric values between
751    /// `ESC[` and `m` in ANSI escape sequences) and converts them into a `Style` struct
752    /// with the appropriate formatting attributes and colors.
753    ///
754    /// # Arguments
755    ///
756    /// * `params` - A slice of u8 values representing SGR codes (e.g., `[1, 31]` for bold red)
757    ///
758    /// # Returns
759    ///
760    /// A `Style` struct with the parsed attributes applied. If the input is empty or
761    /// contains only a reset code (0), returns a default style.
762    ///
763    /// # SGR Code Support
764    ///
765    /// ## Text Attributes
766    /// - `0` - Reset all attributes to default
767    /// - `1` - Bold
768    /// - `2` - Dim
769    /// - `3` - Italic
770    /// - `4` - Underline
771    /// - `5` - Slow blink
772    /// - `6` - Rapid blink
773    /// - `7` - Reverse video
774    /// - `8` - Hidden/concealed
775    /// - `9` - Strikethrough
776    /// - `21` - Double underline
777    /// - `22` - Normal intensity (neither bold nor dim)
778    /// - `23` - Not italic
779    /// - `24` - Not underlined
780    /// - `25` - Not blinking
781    /// - `27` - Not reversed
782    /// - `28` - Not hidden
783    /// - `29` - Not strikethrough
784    ///
785    /// ## Foreground Colors (Basic)
786    /// - `30-37` - Black, Red, Green, Yellow, Blue, Purple, Cyan, White
787    /// - `39` - Default foreground color
788    /// - `90-97` - Bright variants of basic colors
789    ///
790    /// ## Background Colors (Basic)
791    /// - `40-47` - Black, Red, Green, Yellow, Blue, Purple, Cyan, White
792    /// - `49` - Default background color
793    /// - `100-107` - Bright variants of basic colors
794    ///
795    /// ## Extended Colors
796    /// - `38;5;n` - Set foreground to 256-color palette color n
797    /// - `38;2;r;g;b` - Set foreground to RGB color
798    /// - `48;5;n` - Set background to 256-color palette color n
799    /// - `48;2;r;g;b` - Set background to RGB color
800    ///
801    /// ## Fonts
802    /// - `10-19` - Select font (primary or alternate fonts 1-9)
803    /// - `20` - Fraktur font
804    ///
805    /// # Notes
806    ///
807    /// - Unknown or unsupported SGR codes are collected in `style.unknown`
808    /// - Extended color sequences (38/48 with 5 or 2) consume multiple parameters
809    /// - If extended color sequences are incomplete, the codes are stored as unknown
810    /// - Reset code (0) clears all attributes and returns a default style
811    pub fn parse(params: &[u8]) -> AnsiSelectGraphicRendition {
812        let mut style = AnsiSelectGraphicRendition::default();

814

825                22 => style.intensity = Some(Intensity::Normal),

829                23 => style.italic = Some(false),
830
831                // Underline
832                4 => style.underline = Some(Underline::Single),
833                21 => style.underline = Some(Underline::Double),
834                24 => style.underline = Some(Underline::Disabled),
835
836                // Blink
837                5 => style.blink = Some(Blink::Slow),
838                6 => style.blink = Some(Blink::Rapid),
839                25 => style.blink = Some(Blink::Off),
840
841                // Reverse
842                7 => style.reverse = Some(true),
843                27 => style.reverse = Some(false),
844
845                // Hidden
846                8 => style.hidden = Some(true),
847                28 => style.hidden = Some(false),
848
849                // Strike
850                9 => style.strike = Some(true),
851                29 => style.strike = Some(false),
852
853                // Fonts
854                10 => style.font = Some(Font::PrimaryFont),
855                11 => style.font = Some(Font::AlternateFont1),
856                12 => style.font = Some(Font::AlternateFont2),
857                13 => style.font = Some(Font::AlternateFont3),
858                14 => style.font = Some(Font::AlternateFont4),
859                15 => style.font = Some(Font::AlternateFont5),
860                16 => style.font = Some(Font::AlternateFont6),
861                17 => style.font = Some(Font::AlternateFont7),
862                18 => style.font = Some(Font::AlternateFont8),
863                19 => style.font = Some(Font::AlternateFont9),
864                20 => style.font = Some(Font::Fraktur),
865
866                // Foreground colors (basic)
867                30 => style.foreground = Some(Color::Black),
868                31 => style.foreground = Some(Color::Red),
869                32 => style.foreground = Some(Color::Green),
870                33 => style.foreground = Some(Color::Yellow),
871                34 => style.foreground = Some(Color::Blue),
872                35 => style.foreground = Some(Color::Purple),
873                36 => style.foreground = Some(Color::Cyan),
874                37 => style.foreground = Some(Color::White),
875                39 => style.foreground = None, // Default foreground
876
877                // Background colors (basic)
878                40 => style.background = Some(Color::Black),
879                41 => style.background = Some(Color::Red),
880                42 => style.background = Some(Color::Green),
881                43 => style.background = Some(Color::Yellow),
882                44 => style.background = Some(Color::Blue),
883                45 => style.background = Some(Color::Purple),
884                46 => style.background = Some(Color::Cyan),
885                47 => style.background = Some(Color::White),
886                49 => style.background = None, // Default background
887
888                // Bright foreground colors
889                90 => style.foreground = Some(Color::Black),
890                91 => style.foreground = Some(Color::Red),
891                92 => style.foreground = Some(Color::Green),
892                93 => style.foreground = Some(Color::Yellow),
893                94 => style.foreground = Some(Color::Blue),
894                95 => style.foreground = Some(Color::Purple),
895                96 => style.foreground = Some(Color::Cyan),
896                97 => style.foreground = Some(Color::White),
897
898                // Bright background colors
899                100 => style.background = Some(Color::Black),
900                101 => style.background = Some(Color::Red),
901                102 => style.background = Some(Color::Green),
902                103 => style.background = Some(Color::Yellow),
903                104 => style.background = Some(Color::Blue),
904                105 => style.background = Some(Color::Purple),
905                106 => style.background = Some(Color::Cyan),
906                107 => style.background = Some(Color::White),
907
908                // Extended foreground color
909                38 => {
910                    if i + 2 < params.len() && params[i + 1] == 5 {
911                        // 256-color: 38;5;n
912                        style.foreground = Some(Color::Fixed(params[i + 2]));
913                        i += 2;
914                    } else if i + 4 < params.len() && params[i + 1] == 2 {
915                        // RGB: 38;2;r;g;b
916                        style.foreground =
917                            Some(Color::RGB(params[i + 2], params[i + 3], params[i + 4]));
918                        i += 4;
919                    } else {
920                        // Incomplete sequence, store as unknown
921                        style.unknown.push(SGRParameter::Unknown(params[i]));
922                    }
923                }
924
925                // Extended background color
926                48 => {
927                    if i + 2 < params.len() && params[i + 1] == 5 {
928                        // 256-color: 48;5;n
929                        style.background = Some(Color::Fixed(params[i + 2]));
930                        i += 2;
931                    } else if i + 4 < params.len() && params[i + 1] == 2 {
932                        // RGB: 48;2;r;g;b
933                        style.background =
934                            Some(Color::RGB(params[i + 2], params[i + 3], params[i + 4]));
935                        i += 4;
936                    } else {
937                        // Incomplete sequence, store as unknown
938                        style.unknown.push(SGRParameter::Unknown(params[i]));
939                    }
940                }
941
942                // Unknown or unsupported codes
943                _ => {
944                    style.unknown.push(SGRParameter::Unknown(params[i]));
945                }
946            }
947
948            i += 1;
949        }
950
951        style
952    }

1498    /// Converts this color to its nearest equivalent in the basic 16-color palette.
1499    ///
1500    /// This method performs downsampling from higher color modes to the basic ANSI
1501    /// 16-color palette (8 standard colors + 8 bright variants). This is useful when:
1502    ///
1503    /// - Targeting terminals with limited color support
1504    /// - Ensuring maximum compatibility across different terminal emulators
1505    /// - Reducing color complexity while maintaining reasonable visual fidelity
1506    ///
1507    /// # Conversion Rules
1508    ///
1509    /// - **Basic colors** ([`Black`](Color::Black)-[`White`](Color::White)): Returned unchanged
1510    /// - **Bright colors** ([`BrightBlack`](Color::BrightBlack)-[`BrightWhite`](Color::BrightWhite)): Returned unchanged
1511    /// - **Fixed palette** ([`Fixed`](Color::Fixed)): Mapped to the corresponding basic/bright color
1512    ///   - Colors 0-7 map to basic colors
1513    ///   - Colors 8-15 map to bright colors
1514    ///   - Colors 16-255 are converted to their nearest basic color equivalent
1515    /// - **RGB colors** ([`RGB`](Color::RGB)): Mapped to the nearest basic color by calculating
1516    ///   color distance in RGB space
1517    ///
1518    /// # Algorithm
1519    ///
1520    /// For RGB colors, the conversion finds the closest basic color by:
1521    /// 1. Calculating the Euclidean distance in RGB color space
1522    /// 2. Selecting the basic/bright color with the minimum distance
1523    /// 3. Choosing bright variants for colors with higher overall intensity
1524    ///
1525    /// # Examples
1526    ///
1527    /// Basic colors are preserved:
1528    ///
1529    /// ```
1530    /// use termionix_ansicodec::Color;
1531    ///
1532    /// let red = Color::Red;
1533    /// assert_eq!(red.to_basic(), Color::Red);
1534    ///
1535    /// let bright_blue = Color::BrightBlue;
1536    /// assert_eq!(bright_blue.to_basic(), Color::BrightBlue);
1537    /// ```
1538    ///
1539    /// Fixed palette colors are converted:
1540    ///
1541    /// ```
1542    /// use termionix_ansicodec::Color;
1543    ///
1544    /// // Color 0-7 map to basic colors
1545    /// let fixed_red = Color::Fixed(1);
1546    /// assert_eq!(fixed_red.to_basic(), Color::Red);
1547    ///
1548    /// // Color 8-15 map to bright colors
1549    /// let fixed_bright_red = Color::Fixed(9);
1550    /// assert_eq!(fixed_bright_red.to_basic(), Color::BrightRed);
1551    ///
1552    /// // Higher colors are converted to nearest match
1553    /// let fixed_196 = Color::Fixed(196); // Bright red in 256-color palette
1554    /// let basic = fixed_196.to_basic(); // Converts to Red or BrightRed
1555    /// ```
1556    ///
1557    /// RGB colors are approximated:
1558    ///
1559    /// ```
1560    /// use termionix_ansicodec::Color;
1561    ///
1562    /// // Bright red RGB
1563    /// let rgb_red = Color::RGB(255, 0, 0);
1564    /// let basic = rgb_red.to_basic();
1565    /// // Results in Color::Red or Color::BrightRed depending on intensity
1566    ///
1567    /// // Custom orange color
1568    /// let orange = Color::RGB(255, 165, 0);
1569    /// let basic = orange.to_basic();
1570    /// // Approximated to nearest basic color (likely Yellow or BrightYellow)
1571    /// ```
1572    ///
1573    /// # Use Cases
1574    ///
1575    /// Fallback for limited terminals:
1576    ///
1577    /// ```
1578    /// use termionix_ansicodec::{Color, ColorMode};
1579    ///
1580    /// fn get_color_for_mode(color: Color, mode: &ColorMode) -> Color {

1590    /// # Performance
1591    ///
1592    /// - Basic/Bright colors: O(1) - immediate return
1593    /// - Fixed colors: O(1) for colors 0-15, O(1) computation for others
1594    /// - RGB colors: O(1) - fixed number of distance calculations
1595    ///
1596    /// # See Also
1597    ///
1598    /// - [`to_fixed()`](Color::to_fixed) - Convert to 256-color palette
1599    /// - [`to_truecolor()`](Color::to_truecolor) - Convert to 24-bit RGB
1600    /// - [`ColorMode::Basic`] - The color mode that uses this conversion
1601    pub fn to_basic(&self) -> Color {
1602        match self {
1603            Color::Black => Color::Black,
1604            Color::Red => Color::Red,
1605            Color::Green => Color::Green,
1606            Color::Yellow => Color::Yellow,
1607            Color::Blue => Color::Blue,
1608            Color::Purple => Color::Purple,
1609            Color::Cyan => Color::Cyan,
1610            Color::White => Color::White,
1611
1612            Color::BrightBlack => Color::BrightBlack,
1613            Color::BrightRed => Color::BrightRed,
1614            Color::BrightGreen => Color::BrightGreen,
1615            Color::BrightYellow => Color::BrightYellow,
1616            Color::BrightBlue => Color::BrightBlue,
1617            Color::BrightPurple => Color::BrightPurple,
1618            Color::BrightCyan => Color::BrightCyan,
1619            Color::BrightWhite => Color::BrightWhite,
1620
1621            Color::Fixed(n) => {
1622                if *n < 8 {
1623                    // Colors 0-7 map to basic colors
1624                    match n {
1625                        0 => Color::Black,
1626                        1 => Color::Red,
1627                        2 => Color::Green,
1628                        3 => Color::Yellow,
1629                        4 => Color::Blue,
1630                        5 => Color::Purple,
1631                        6 => Color::Cyan,
1632                        7 => Color::White,
1633                        _ => unreachable!(),
1634                    }
1635                } else if *n < 16 {
1636                    // Colors 8-15 map to bright colors
1637                    match n {
1638                        8 => Color::BrightBlack,
1639                        9 => Color::BrightRed,
1640                        10 => Color::BrightGreen,
1641                        11 => Color::BrightYellow,
1642                        12 => Color::BrightBlue,
1643                        13 => Color::BrightPurple,
1644                        14 => Color::BrightCyan,
1645                        15 => Color::BrightWhite,
1646                        _ => unreachable!(),
1647                    }
1648                } else {
1649                    // For other fixed colors, approximate to nearest basic color
1650                    Self::rgb_to_basic(
1651                        (((*n as u16 - 16) / 36) * 51) as u8,
1652                        ((((*n as u16 - 16) % 36) / 6) * 51) as u8,
1653                        (((*n as u16 - 16) % 6) * 51) as u8,
1654                    )
1655                }
1656            }
1657
1658            Color::RGB(r, g, b) => Self::rgb_to_basic(*r, *g, *b),
1659        }
1660    }

1662    /// Converts this color to the 256-color fixed palette format.
1663    ///
1664    /// This method converts any color representation to a [`Fixed`](Color::Fixed) palette
1665    /// index (0-255). The 256-color palette provides a good balance between color variety
1666    /// and terminal compatibility, supported by most modern terminal emulators.
1667    ///
1668    /// # Conversion Rules
1669    ///
1670    /// - **Basic colors** ([`Black`](Color::Black)-[`White`](Color::White)): Mapped to palette indices 0-7
1671    /// - **Bright colors** ([`BrightBlack`](Color::BrightBlack)-[`BrightWhite`](Color::BrightWhite)): Mapped to palette indices 8-15
1672    /// - **Fixed palette** ([`Fixed`](Color::Fixed)): Returned unchanged
1673    /// - **RGB colors** ([`RGB`](Color::RGB)): Mapped to the nearest color in the 256-color palette
1674    ///
1675    /// # 256-Color Palette Structure
1676    ///
1677    /// The fixed palette is organized as follows:
1678    ///
1679    /// - **0-7**: Basic ANSI colors (Black, Red, Green, Yellow, Blue, Purple, Cyan, White)
1680    /// - **8-15**: Bright ANSI colors (corresponding bright variants)
1681    /// - **16-231**: 666 RGB cube with formula: `16 + 36r + 6g + b` where r,g,b  [0,5]
1682    /// - **232-255**: 24-level grayscale ramp from dark to light
1683    ///
1684    /// # RGB to Fixed Conversion Algorithm
1685    ///
1686    /// For RGB colors, the conversion process:
1687    /// 1. Determines if the color is a grayscale (R  G  B)
1688    /// 2. For grayscale: Maps to the grayscale ramp (indices 232-255) or extreme black/white
1689    /// 3. For color: Converts to the 666 RGB cube (indices 16-231) by:
1690    ///    - Quantizing each RGB channel from 0-255 to 0-5
1691    ///    - Applying the formula: `16 + 36r + 6g + b`
1692    ///

1704    ///

1711    /// assert_eq!(Color::BrightRed.to_fixed(), Color::Fixed(9));

1721    /// let red = Color::RGB(255, 0, 0);
1722    /// let fixed = red.to_fixed();
1723    /// // Results in a Color::Fixed variant close to red
1724    ///
1725    /// // Grayscale values map to grayscale ramp

1961    pub fn to_truecolor(&self) -> Color {
1962        match self {
1963            Color::Black => Color::RGB(0, 0, 0),
1964            Color::Red => Color::RGB(205, 0, 0),
1965            Color::Green => Color::RGB(0, 205, 0),
1966            Color::Yellow => Color::RGB(205, 205, 0),
1967            Color::Blue => Color::RGB(0, 0, 238),
1968            Color::Purple => Color::RGB(205, 0, 205),
1969            Color::Cyan => Color::RGB(0, 205, 205),
1970            Color::White => Color::RGB(229, 229, 229),
1971
1972            Color::BrightBlack => Color::RGB(127, 127, 127),
1973            Color::BrightRed => Color::RGB(255, 0, 0),
1974            Color::BrightGreen => Color::RGB(0, 255, 0),
1975            Color::BrightYellow => Color::RGB(255, 255, 0),
1976            Color::BrightBlue => Color::RGB(92, 92, 255),
1977            Color::BrightPurple => Color::RGB(255, 0, 255),
1978            Color::BrightCyan => Color::RGB(0, 255, 255),
1979            Color::BrightWhite => Color::RGB(255, 255, 255),
1980
1981            Color::Fixed(n) => {
1982                if *n < 16 {
1983                    // Use the basic/bright color mappings
1984                    match n {
1985                        0 => Color::RGB(0, 0, 0),
1986                        1 => Color::RGB(205, 0, 0),
1987                        2 => Color::RGB(0, 205, 0),
1988                        3 => Color::RGB(205, 205, 0),
1989                        4 => Color::RGB(0, 0, 238),
1990                        5 => Color::RGB(205, 0, 205),
1991                        6 => Color::RGB(0, 205, 205),
1992                        7 => Color::RGB(229, 229, 229),
1993                        8 => Color::RGB(127, 127, 127),
1994                        9 => Color::RGB(255, 0, 0),
1995                        10 => Color::RGB(0, 255, 0),
1996                        11 => Color::RGB(255, 255, 0),
1997                        12 => Color::RGB(92, 92, 255),
1998                        13 => Color::RGB(255, 0, 255),
1999                        14 => Color::RGB(0, 255, 255),
2000                        15 => Color::RGB(255, 255, 255),
2001                        _ => unreachable!(),
2002                    }
2003                } else if *n < 232 {
2004                    // 216-color cube (16-231)
2005                    let idx = *n as u16 - 16;
2006                    let r = ((idx / 36) * 51) as u8;
2007                    let g = (((idx % 36) / 6) * 51) as u8;
2008                    let b = ((idx % 6) * 51) as u8;
2009                    Color::RGB(r, g, b)
2010                } else {
2011                    // Grayscale (232-255)
2012                    let gray = ((*n as u16 - 232) * 10 + 8) as u8;
2013                    Color::RGB(gray, gray, gray)
2014                }
2015            }
2016
2017            Color::RGB(r, g, b) => Color::RGB(*r, *g, *b),
2018        }
2019    }

2021    /// Converts RGB values to the nearest basic 16-color palette color.
2022    ///
2023    /// This is an internal helper method used by [`to_basic()`](Color::to_basic) to perform
2024    /// the actual RGB-to-basic color conversion. It calculates the Euclidean distance in
2025    /// RGB color space to find the closest match from the 16 basic/bright ANSI colors.
2026    ///
2027    /// # Arguments
2028    ///
2029    /// * `r` - Red channel value (0-255)
2030    /// * `g` - Green channel value (0-255)
2031    /// * `b` - Blue channel value (0-255)
2032    ///
2033    /// # Algorithm
2034    ///
2035    /// 1. Computes the distance to each of the 16 basic colors using the formula:
2036    ///    ```text
2037    ///    distance = ((r-r) + (g-g) + (b-b))
2038    ///    ```
2039    /// 2. Returns the color with the minimum distance
2040    /// 3. Prioritizes bright variants for higher-intensity colors
2041    ///
2042    /// # Returns
2043    ///
2044    /// A [`Color`] variant from the basic 16-color palette that most closely matches
2045    /// the input RGB values.
2046    ///
2047    /// # Examples
2048    ///
2049    /// This method is primarily used internally:
2050    ///
2051    /// ```ignore
2052    /// // Internal usage in to_basic()
2053    /// let rgb_red = Color::RGB(255, 0, 0);
2054    /// // Internally calls rgb_to_basic(255, 0, 0)
2055    /// let basic = rgb_red.to_basic();
2056    /// ```
2057    ///
2058    /// # Performance
2059    ///
2060    /// O(1) - Fixed number of distance calculations (16 colors)
2061    ///
2062    /// # See Also
2063    ///
2064    /// - [`to_basic()`](Color::to_basic) - Public method that uses this helper
2065    /// - [`rgb_to_fixed_index()`](Color::rgb_to_fixed_index) - Similar conversion for 256-color palette
2066    fn rgb_to_basic(r: u8, g: u8, b: u8) -> Color {
2067        // Calculate perceived brightness
2068        let brightness = (r as u32 * 299 + g as u32 * 587 + b as u32 * 114) / 1000;
2069
2070        // Determine which color component is dominant
2071        let max = r.max(g).max(b);
2072        let min = r.min(g).min(b);
2073        let saturation = if max > 0 {
2074            ((max - min) as f32 / max as f32) * 100.0
2075        } else {
2076            0.0
2077        };
2078
2079        // Low saturation means grayscale
2080        if saturation < 25.0 {
2081            if brightness > 127 {
2082                return Color::BrightWhite;
2083            } else if brightness > 64 {
2084                return Color::White;
2085            } else if brightness > 32 {
2086                return Color::BrightBlack;
2087            } else {
2088                return Color::Black;
2089            }
2090        }
2091
2092        // For saturated colors, choose based on dominant component
2093        let is_bright = brightness > 127;
2094
2095        if r > g && r > b {
2096            // Red dominant
2097            if is_bright {
2098                Color::BrightRed
2099            } else {
2100                Color::Red
2101            }
2102        } else if g > r && g > b {
2103            // Green dominant
2104            if is_bright {
2105                Color::BrightGreen
2106            } else {
2107                Color::Green
2108            }
2109        } else if b > r && b > g {
2110            // Blue dominant
2111            if is_bright {
2112                Color::BrightBlue
2113            } else {
2114                Color::Blue
2115            }
2116        } else if r > 0 && g > 0 && b == min {
2117            // Yellow (red + green)
2118            if is_bright {
2119                Color::BrightYellow
2120            } else {
2121                Color::Yellow
2122            }
2123        } else if r > 0 && b > 0 && g == min {
2124            // Magenta/Purple (red + blue)
2125            if is_bright {
2126                Color::BrightPurple
2127            } else {
2128                Color::Purple
2129            }
2130        } else if g > 0 && b > 0 && r == min {
2131            // Cyan (green + blue)
2132            if is_bright {
2133                Color::BrightCyan
2134            } else {
2135                Color::Cyan
2136            }
2137        } else {
2138            // Fallback to white/black based on brightness
2139            if is_bright {
2140                Color::BrightWhite
2141            } else {
2142                Color::Black
2143            }
2144        }
2145    }

2147    /// Converts RGB values to the nearest 256-color palette index.
2148    ///
2149    /// This is an internal helper method used by [`to_fixed()`](Color::to_fixed) to convert
2150    /// RGB colors to the 256-color fixed palette. It intelligently determines whether to
2151    /// use the grayscale ramp or the RGB cube based on the color characteristics.
2152    ///
2153    /// # Arguments
2154    ///
2155    /// * `r` - Red channel value (0-255)
2156    /// * `g` - Green channel value (0-255)
2157    /// * `b` - Blue channel value (0-255)
2158    ///
2159    /// # Algorithm
2160    ///
2161    /// 1. **Grayscale Detection**: If R  G  B (within a threshold), maps to grayscale ramp
2162    ///    - Pure black (0,0,0)  Index 0
2163    ///    - Pure white (255,255,255)  Index 15
2164    ///    - Gray values  Indices 232-255 (24-level grayscale ramp)
2165    ///
2166    /// 2. **Color Mapping**: Otherwise, maps to the 666 RGB cube (indices 16-231)
2167    ///    - Quantizes each RGB channel from 0-255 to 0-5
2168    ///    - Applies formula: `16 + 36r + 6g + b`
2169    ///
2170    /// # Quantization
2171    ///
2172    /// RGB values are quantized to 6 levels using:
2173    /// ```text
2174    /// level = round(channel / 255  5)
2175    /// ```
2176    ///
2177    /// This maps:
2178    /// - 0-25  0
2179    /// - 26-76  1
2180    /// - 77-127  2
2181    /// - 128-178  3
2182    /// - 179-229  4
2183    /// - 230-255  5
2184    ///
2185    /// # Returns
2186    ///
2187    /// A palette index (0-255) representing the closest color in the 256-color palette.
2188    ///
2189    /// # Examples
2190    ///
2191    /// This method is primarily used internally:
2192    ///
2193    /// ```ignore
2194    /// // Internal usage in to_fixed()
2195    /// let rgb_color = Color::RGB(128, 64, 192);
2196    /// // Internally calls rgb_to_fixed_index(128, 64, 192)
2197    /// let fixed = rgb_color.to_fixed();
2198    /// ```
2199    ///
2200    /// # Color Space Mapping
2201    ///
2202    /// The 256-color palette structure:
2203    ///
2204    /// ```text
2205    /// 0-15:   Basic/Bright colors (standard ANSI)
2206    /// 16-231: 666 RGB cube (216 colors)
2207    ///         16 + 36r + 6g + b where r,g,b  [0,5]
2208    /// 232-255: Grayscale ramp (24 shades)
2209    /// ```
2210    ///
2211    /// # Performance
2212    ///
2213    /// O(1) - Constant-time calculation with no loops
2214    ///
2215    /// # See Also
2216    ///
2217    /// - [`to_fixed()`](Color::to_fixed) - Public method that uses this helper
2218    /// - [`rgb_to_basic()`](Color::rgb_to_basic) - Similar conversion for basic colors
2219    /// - [XTerm 256 Color Chart](https://upload.wikimedia.org/wikipedia/commons/1/15/Xterm_256color_chart.svg)
2220    fn rgb_to_fixed_index(r: u8, g: u8, b: u8) -> u8 {
2221        // Check if it's close to a grayscale value
2222        let avg = (r as u16 + g as u16 + b as u16) / 3;
2223        let variance = ((r as i16 - avg as i16).abs()
2224            + (g as i16 - avg as i16).abs()
2225            + (b as i16 - avg as i16).abs()) as u16;
2226
2227        if variance < 15 {
2228            // Map to grayscale colors (232-255)
2229            if avg < 8 {
2230                return 16; // Use color cube black
2231            } else if avg > 238 {
2232                return 231; // Use color cube white
2233            } else {
2234                return (232 + (avg - 8) / 10) as u8;
2235            }
2236        }
2237
2238        // Map to 216-color cube (16-231)
2239        let r_idx = ((r as u16 * 5 + 127) / 255) as u8;
2240        let g_idx = ((g as u16 * 5 + 127) / 255) as u8;
2241        let b_idx = ((b as u16 * 5 + 127) / 255) as u8;
2242
2243        16 + 36 * r_idx + 6 * g_idx + b_idx
2244    }
2245}

2248///

2250/// alternate fonts, including a special Fraktur (gothic) font. Font support varies

2268/// | `AlternateFont8` | 18 | Eighth alternate font |

3559/// - When unsupported, terminals typically ignore these codes without error

4018/// SGR (Select Graphic Rendition) Parameter
4019///
4020/// Represents additional ANSI escape sequence SGR parameters that control text
4021/// rendering attributes beyond the basic style properties like intensity, italic,
4022/// underline, etc. These are less commonly used SGR codes that don't fit into
4023/// the main `Style` struct fields.
4024///
4025/// # SGR Codes
4026///
4027/// SGR codes are part of ANSI escape sequences in the format `CSI n m` where
4028/// `n` is the numeric parameter. This enum represents SGR codes that are not
4029/// covered by the primary `Style` struct fields.
4030///
4031/// # Variants
4032///
4033/// ## Color Reset
4034/// - `DefaultForegroundColor` - Resets foreground color to terminal default (SGR 39)
4035/// - `DefaultBackgroundColor` - Resets background color to terminal default (SGR 49)
4036///
4037/// ## Text Effects
4038/// - `Framed` - Adds a frame around text (SGR 51)
4039/// - `Encircled` - Draws a circle around text (SGR 52)
4040/// - `Overlined` - Draws a line above text (SGR 53)
4041/// - `NotFramedNotEncircled` - Removes frame or encircle effects (SGR 54)
4042/// - `NotOverlined` - Removes overline effect (SGR 55)
4043///
4044/// ## Spacing
4045/// - `DisableProportionalSpacing` - Disables proportional character spacing (SGR 50)
4046///
4047/// ## Underline Color
4048/// - `SetUnderlineColor` - Sets the color of underlines independently from text color (SGR 58)
4049/// - `DefaultUnderlineColor` - Resets underline color to default (SGR 59)
4050///
4051/// ## Unknown
4052/// - `Unknown` - Represents unrecognized or unsupported SGR parameters
4053///
4054/// # Terminal Support
4055///
4056/// Note that many of these parameters have limited support across different terminal
4057/// emulators. The most widely supported are the color reset parameters
4058/// (`DefaultForegroundColor` and `DefaultBackgroundColor`). Effects like `Framed`,
4059/// `Encircled`, and `Overlined` are rarely implemented.
4060///
4061/// # Examples
4062///
4063/// ```rust
4064/// use termionix_ansicodec::{SGRParameter, Color};
4065///
4066/// // Reset foreground to default
4067/// let param = SGRParameter::DefaultForegroundColor;
4068///
4069/// // Set a custom underline color
4070/// let underline_param = SGRParameter::SetUnderlineColor(Color::Red);
4071///
4072/// // Handle unknown parameter
4073/// let unknown = SGRParameter::Unknown(99);
4074/// ```
