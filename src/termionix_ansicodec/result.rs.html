17//! Error types for the ansicodec crate.
18//!
19//! This module provides comprehensive error handling for ANSI string operations,
20//! including parsing errors, validation errors, and buffer management errors.
21
22/// Result type alias for operations that may fail with an [`AnsiCodecError`].
23pub type AnsiCodecResult<T> = Result<T, AnsiCodecError>;
24
25/// Errors that can occur when working with ANSI strings.
26#[derive(Debug)]
27pub enum AnsiCodecError {
28    /// IO Error
29    IoError(std::io::Error),
30    /// Invalid UTF-8 sequence encountered at the specified position.
31    ///
32    /// This error occurs when the input contains bytes that don't form valid UTF-8.
33    InvalidUtf8 {
34        /// The byte position where the invalid UTF-8 was encountered
35        position: usize,
36    },
37
38    /// Malformed ANSI escape sequence encountered.
39    ///
40    /// This error occurs when an ANSI sequence doesn't follow the expected format.
41    MalformedAnsi {
42        /// The byte position where the malformed sequence starts
43        position: usize,
44        /// Description of what's wrong with the sequence
45        description: String,
46    },
47
48    /// Incomplete ANSI sequence at the end of input.
49    ///
50    /// This error occurs when the input ends in the middle of an ANSI sequence.
51    IncompleteSequence {
52        /// The byte position where the incomplete sequence starts
53        position: usize,
54    },
55
56    /// Range is out of bounds for the string.
57    ///
58    /// This error occurs when trying to apply a style or operation to a range
59    /// that extends beyond the string's length.
60    RangeOutOfBounds {
61        /// The range that was requested
62        range: std::ops::Range<usize>,
63        /// The maximum valid position
64        max: usize,
65    },
66
67    /// ANSI sequence exceeds maximum allowed length.
68    ///
69    /// This error occurs when an ANSI sequence is longer than the configured limit,
70    /// which may indicate malformed input or a potential attack.
71    SequenceTooLong {
72        /// The actual length of the sequence
73        length: usize,
74        /// The maximum allowed length
75        max: usize,
76    },
77
78    /// Buffer overflow prevented.
79    ///
80    /// This error occurs when an operation would exceed the buffer's capacity.
81    BufferOverflow {
82        /// The number of bytes attempted to write
83        attempted: usize,
84        /// The buffer's capacity
85        capacity: usize,
86    },
87
88    /// Invalid parameter value.
89    ///
90    /// This error occurs when a parameter has an invalid value.
91    InvalidParameter {
92        /// Name of the parameter
93        name: String,
94        /// The invalid value
95        value: String,
96        /// Description of why it's invalid
97        reason: String,
98    },
99}
100
101impl std::fmt::Display for AnsiCodecError {
102    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
103        match self {
104            AnsiCodecError::IoError(err) => {
105                write!(f, "IOError {}", err)
106            }
107            AnsiCodecError::InvalidUtf8 { position } => {
108                write!(f, "Invalid UTF-8 sequence at position {}", position)
109            }
110            AnsiCodecError::MalformedAnsi {
111                position,
112                description,
113            } => {
114                write!(
115                    f,
116                    "Malformed ANSI sequence at position {}: {}",
117                    position, description
118                )
119            }
120            AnsiCodecError::IncompleteSequence { position } => {
121                write!(
122                    f,
123                    "Incomplete ANSI sequence at end of input (started at position {})",
124                    position
125                )
126            }
127            AnsiCodecError::RangeOutOfBounds { range, max } => {
128                write!(
129                    f,
130                    "Range out of bounds: {:?} (maximum valid position: {})",
131                    range, max
132                )
133            }
134            AnsiCodecError::SequenceTooLong { length, max } => {
135                write!(
136                    f,
137                    "Sequence too long: {} bytes (maximum allowed: {})",
138                    length, max
139                )
140            }
141            AnsiCodecError::BufferOverflow {
142                attempted,
143                capacity,
144            } => {
145                write!(
146                    f,
147                    "Buffer overflow: attempted to write {} bytes to buffer of size {}",
148                    attempted, capacity
149                )
150            }
151            AnsiCodecError::InvalidParameter {
152                name,
153                value,
154                reason,
155            } => {
156                write!(
157                    f,
158                    "Invalid parameter '{}' with value '{}': {}",
159                    name, value, reason
160                )
161            }
162        }
163    }
164}
165
166impl std::error::Error for AnsiCodecError {}
167
168impl From<std::io::Error> for AnsiCodecError {
169    fn from(error: std::io::Error) -> Self {
170        AnsiCodecError::IoError(error)
171    }
172}
173
174impl From<termionix_telnetcodec::TelnetCodecError> for AnsiCodecError {
175    fn from(error: termionix_telnetcodec::TelnetCodecError) -> Self {
176        AnsiCodecError::IoError(std::io::Error::new(
177            std::io::ErrorKind::Other,
178            format!("Codec error: {:?}", error),
179        ))
180    }
181}
