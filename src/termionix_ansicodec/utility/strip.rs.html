17use std::borrow::Cow;
18
19/// Removes ANSI escape sequences from a string.
20///
21/// This function strips ANSI Control Sequence Introducer (CSI) codes from the input string,
22/// which are commonly used for terminal formatting, colors, and cursor control. The function
23/// detects sequences that begin with the escape character (`\x1b`) followed by `[` and removes
24/// them along with their parameters and terminal character.
25///
26/// # ANSI Escape Sequence Format
27///
28/// ANSI CSI sequences follow the pattern: `ESC [ <parameters> <command>`
29/// - `ESC`: The escape character (`\x1b` or decimal 27)
30/// - `[`: Opening square bracket (Control Sequence Introducer)
31/// - `<parameters>`: Optional numeric parameters separated by semicolons
32/// - `<command>`: A letter (A-Z, a-z) or specific symbol (like 'm') that terminates the sequence
33///
34/// # Performance
35///
36/// The function performs a quick check for the presence of the escape character before
37/// processing. If no ANSI codes are found, it returns a borrowed reference to the original
38/// string (zero-copy). Otherwise, it allocates a new `String` with the codes removed.
39///
40/// # Arguments
41///
42/// * `str` - A string slice that may contain ANSI escape sequences
43///
44/// # Returns
45///
46/// Returns a `Cow<'_, str>`:
47/// - `Cow::Borrowed(str)` if the input contains no ANSI escape sequences
48/// - `Cow::Owned(String)` if ANSI codes were found and removed
49///
50/// # Examples
51///
52/// ```
53/// use std::borrow::Cow;
54/// # use termionix_ansicodec::strip_ansi_codes;
55///
56/// // String with ANSI color codes
57/// let colored = "\x1b[1;31mRed Text\x1b[0m";
58/// let clean = strip_ansi_codes(colored);
59/// assert_eq!(clean, "Red Text");
60///
61/// // String without ANSI codes (zero-copy)
62/// let plain = "Plain Text";
63/// let result = strip_ansi_codes(plain);
64/// assert!(matches!(result, Cow::Borrowed(_)));
65/// assert_eq!(result, "Plain Text");
66///
67/// // Complex formatting
68/// let formatted = "\x1b[1mBold\x1b[0m and \x1b[4mUnderlined\x1b[0m";
69/// assert_eq!(strip_ansi_codes(formatted), "Bold and Underlined");
70/// ```
71///
72/// # Supported Sequences
73///
74/// This function handles CSI sequences (ESC `[`) which include:
75/// - Color codes (foreground/background)
76/// - Text styling (bold, italic, underline, etc.)
77/// - Cursor positioning and movement
78/// - Screen clearing and erasing
79///
80/// # Limitations
81///
82/// Currently only strips CSI sequences (ESC `[`). Does not remove:
83/// - OSC sequences (ESC `]`) - Operating System Commands
84/// - DCS sequences (ESC `P`) - Device Control Strings
85/// - Other escape sequences that don't use the `[` delimiter
86///
87/// # See Also
88///
89/// For detailed information about ANSI escape codes, see the ANSI specification
90/// or the project's `ansi.md` documentation file.
91pub fn strip_ansi_codes(str: &str) -> Cow<'_, str> {
92    // Check if the string contains any ANSI escape sequences
93    if !str.contains('\x1b') {
94        // No ANSI codes, return borrowed string
95        return Cow::Borrowed(str);
96    }
97
98    let mut result = String::with_capacity(str.len());
99    let chars: Vec<char> = str.chars().collect();
100    let mut i = 0;
101
102    while i < chars.len() {
103        // Check for ANSI escape sequence start
104        if chars[i] == '\x1b' && i + 1 < chars.len() && chars[i + 1] == '[' {
105            // Skip the escape sequence
106            i += 2; // Skip '\x1b['
107
108            // Skip until we find the terminal character (typically 'm', but could be others)
109            while i < chars.len() {
110                let ch = chars[i];
111                i += 1;
112                // ANSI escape sequences end with a letter (A-Z, a-z) or certain symbols
113                if ch.is_ascii_alphabetic() || ch == 'm' {
114                    break;
115                }
116            }
117        } else {
118            // Regular character, add to result
119            result.push(chars[i]);
120            i += 1;
121        }
122    }
123
124    Cow::Owned(result)
125}
